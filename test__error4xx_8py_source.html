191 # Connect first client socket and send half of the request
192 client1.connect((host, port))

194
195 # Wait a bit to ensure the next parts trigger epoll simultaneously
196 time.sleep(0.1)
197
198 # Send the second half of the first request and create second client socket
199 client1.sendall(b"HTTP/1.1\r\n\r\n")
200 client2.connect((host, port))
201
202 # And send full new request
203 client2.sendall(b"GET / HTTP/1.1\r\nHost: localhost\r\n\r\n")
204
205 # Read responses
