123void ResponseBodyHandler::parseCGIResponseHeaders()
124{
125 LOG_DEBUG << "Parsing received CGI response...";
126
127 const size_t sizeCRLF = 2;
128 const size_t sizeCRLFCRLF = 4;
129 const size_t posHeadersEnd = m_responseBody.find("\r\n\r\n");
130 // Include one CRLF at the end of last header line
131 std::string headers = m_responseBody.substr(0, posHeadersEnd + sizeCRLF);
132 std::string loweredHeaders = headers;
133 webutils::lowercase(loweredHeaders);
134
135 if (posHeadersEnd == std::string::npos) {
136 m_request.httpStatus = StatusInternalServerError;
137 handleErrorBody();
138 LOG_ERROR << ERR_MISSING_CGI_HEADER;
139 return;
140 }
141
142 if (loweredHeaders.find("content-type: ") == std::string::npos
143 && loweredHeaders.find("location: ") == std::string::npos
144 && loweredHeaders.find("status: ") == std::string::npos) {
145 m_request.httpStatus = StatusInternalServerError;
146 handleErrorBody();
147 LOG_ERROR << ERR_MISSING_CGI_FIELD;
148 return;
149 }
150
151 size_t lineStart = 0;
152 size_t lineEnd = headers.find("\r\n");
153
154 while (lineEnd != std::string::npos) {
155 std::string header = headers.substr(lineStart, lineEnd - lineStart);
156 const std::size_t delimiterPos = header.find_first_of(':');
157 if (delimiterPos != std::string::npos) {
158 std::string headerName = header.substr(0, delimiterPos);
159 webutils::lowercase(headerName);
160 std::string headerValue = header.substr(delimiterPos + 1);
161 headerValue = webutils::trimLeadingWhitespaces(headerValue);
162 webutils::trimTrailingWhiteSpaces(headerValue);
163 m_responseHeaders[headerName] = headerValue;
164 LOG_DEBUG << "Parsed response header: " << headerName << " -> " << headerValue;
165 }
166
167 lineStart = lineEnd + 2;
168 lineEnd = headers.find("\r\n", lineStart);
169 }
170 m_responseBody.erase(0, posHeadersEnd + sizeCRLFCRLF);
171 validateCGIResponseHeaders();
172}

180void ResponseBodyHandler::validateCGIResponseHeaders()
181{
182 // Status
183 std::map<std::string, std::string>::iterator iter = m_responseHeaders.find("status");
184 if (iter != m_responseHeaders.end()) {
185 m_request.httpStatus = extractStatusCode(iter->second);
186 if (m_request.httpStatus == NoStatus) {
187 m_request.httpStatus = StatusInternalServerError;
188 handleErrorBody();
189 m_responseHeaders.clear();
190 LOG_ERROR << "Invalid Status header value encountered in CGI response";
191 return;
192 }
193 }
194
195 // Connection
196 iter = m_responseHeaders.find("connection");
197 if (iter != m_responseHeaders.end()) {
198 if (iter->second != "keep-alive" && iter->second != "close") {
199 m_request.httpStatus = StatusInternalServerError;
200 handleErrorBody();
201 m_responseHeaders.clear();
202 LOG_ERROR << "Invalid Connection header value: " << iter->second;
203 return;
204 }
205
206 if (iter->second == "close")
207 m_request.shallCloseConnection = true;
208 m_responseHeaders.erase(iter);
209 }
