9def parse_http_response(sock):
10 # Receive the full response
11 response_data = sock.recv(4096)
12
13 # Decode response
14 response_str = response_data.decode('utf-8')
15
16 # Split into lines
17 response_lines = response_str.split('\r\n')
18
19 # Status line (first line)
20 status_line = response_lines[0]
21 status_code = int(status_line.split()[1])
22
23 # Headers (remaining lines)
24 headers = {}
25 for line in response_lines[1:]:
26 if line.strip() == "":
27 # End of headers
28 break
29 key, value = line.split(":", 1)
30 headers[key.strip()] = value.strip()
31
32 return {
33 "status_code": status_code,

35 "body": response_str.split('\r\n\r\n', 1)[1] if '\r\n\r\n' in response_str else ""
