48const ConfigFile& ConfigFileParser::parseConfigFile(const std::string& configFilePath)
49{
50 std::ifstream fileStream(configFilePath.c_str());
51 std::stringstream bufferStream;
52
53 if (!fileStream.is_open())
54 throw std::runtime_error(ERR_CONFIG_FILE_OPEN_FAILED);
55 if (fileStream.peek() == std::ifstream::traits_type::eof())
56 throw std::runtime_error(ERR_CONFIG_FILE_EMPTY);
57
58 bufferStream << fileStream.rdbuf();
59 m_configFileContent = bufferStream.str();
60 fileStream.close();
61
62 if (isBracketOpen())
63 throw std::runtime_error(ERR_OPEN_BRACKET_IN_CONFIG_FILE);
64
65 if (!isValidBlockBegin(HttpBlock))
66 throw std::runtime_error(ERR_MISSING_HTTP_BLOCK);
67
68 skipBlockBegin(HttpBlock);
69
70 while (m_configFileContent[m_configFileIndex] != '}') {
71 if (isValidBlockBegin(ServerBlock))
72 readServerBlock();
73 else if (std::isspace(m_configFileContent[m_configFileIndex]) == 0)
74 throw std::runtime_error(ERR_INVALID_DIRECTIVE);
75 m_configFileIndex++;
76 }
77
78 if (m_serverBlocksConfig.empty())
79 throw std::runtime_error(ERR_MISSING_SERVER_BLOCK);
80
81 for (std::vector<ServerBlockConfig>::const_iterator serverIt = m_serverBlocksConfig.begin();
82 serverIt != m_serverBlocksConfig.end(); serverIt++) {
83 processServerContent(*serverIt);
84 m_serverIndex++;
85 }
86
87 if (isLocationDuplicate())
88 throw std::runtime_error(ERR_DUPLICATE_LOCATION);
89
90 return m_configFile;
91}

118{

121 while (std::isspace(m_configFileContent[index]) != 0)

123
124 if (block == HttpBlock && !isKeyword(convertBlockToString(HttpBlock), index))

126 if (block == ServerBlock && !isKeyword(convertBlockToString(ServerBlock), index))

516void ConfigFileParser::readListen(const std::string& value)
517{
518 const size_t colonIndex = value.find(':');
519 const size_t endIndex = value.length();
520 const size_t dot = value.find('.');
521
522 if (colonIndex != std::string::npos) {
523 std::string ipAddress = value.substr(0, colonIndex);
524 if (!webutils::isIpAddressValid(ipAddress))
525 throw std::runtime_error(ERR_INVALID_IP_ADDRESS);
526
527 if (ipAddress == "localhost")
528 m_configFile.servers[m_serverIndex].host = "127.0.0.1";
529 else
530 m_configFile.servers[m_serverIndex].host = ipAddress;
531
532 std::string port = value.substr(colonIndex + 1, endIndex - colonIndex - 1);
533 if (port.find_first_of(s_whitespace) != std::string::npos)
534 throw std::runtime_error(ERR_INVALID_LISTEN_PARAMETERS);
535 if (!webutils::isPortValid(port))
536 throw std::runtime_error(ERR_INVALID_PORT);
537 m_configFile.servers[m_serverIndex].port = port;
538 return;
539 }
540
541 if (dot == std::string::npos) {
542 std::string hostOrPort = value.substr(0, endIndex);
543 if (hostOrPort.find_first_of(s_whitespace) != std::string::npos)
544 throw std::runtime_error(ERR_INVALID_LISTEN_PARAMETERS);
545 if (hostOrPort.empty())
546 throw std::runtime_error(ERR_EMPTY_LISTEN_VALUE);
547 if (hostOrPort == "localhost")
548 m_configFile.servers[m_serverIndex].host = "127.0.0.1";
549 else {
550 if (!webutils::isPortValid(hostOrPort))
551 throw std::runtime_error(ERR_INVALID_PORT);
552 m_configFile.servers[m_serverIndex].port = hostOrPort;
553 }
554 return;
555 }
556
557 std::string ipAddress = value.substr(0, endIndex);
558 if (ipAddress.find_first_of(s_whitespace) != std::string::npos)
559 throw std::runtime_error(ERR_INVALID_LISTEN_PARAMETERS);
560 if (!webutils::isIpAddressValid(ipAddress))
561 throw std::runtime_error(ERR_INVALID_IP_ADDRESS);

589void ConfigFileParser::readMaxBodySize(const Block& block, const std::string& maxBodySize)
590{
591 if (maxBodySize.find_first_of(s_whitespace) != std::string::npos)
592 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_PARAMETERS);
593
594 const size_t lastNumberIndex = maxBodySize.find_last_of(s_number);
595 if (lastNumberIndex == std::string::npos)
596 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_VALUE);
597
598 const std::string number = maxBodySize.substr(0, lastNumberIndex + 1);
599 if (number.find_first_not_of(s_number) != std::string::npos)
600 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_VALUE);
601
602 errno = 0;
603 size_t size = std::strtoul(number.c_str(), NULL, constants::g_decimalBase);
604 if (errno == ERANGE)
605 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_NUMBER_OVERFLOW);
606
607 if (lastNumberIndex != maxBodySize.size() - 1) {
608 const std::string letter = maxBodySize.substr(lastNumberIndex + 1);
609 if (letter.length() != 1)
610 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_UNIT);
611
612 size_t unit = 1;
613 switch (letter.at(0)) {
614 case 'k':
615 case 'K':
616 unit = constants::g_oneKilobyte;
617 break;
618 case 'm':
619 case 'M':
620 unit = constants::g_oneMegabyte;
621 break;
622 case 'g':
623 case 'G':
624 unit = constants::g_oneGigabyte;
625 break;
626 default:
627 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_UNIT);
628 }
629
630 if (size > std::numeric_limits<size_t>::max() / unit)
631 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_UNIT_OVERFLOW);
632
633 size *= unit;
634 }
635
636 if (block == ServerBlock)
637 m_configFile.servers[m_serverIndex].maxBodySize = size;
638 else if (block == LocationBlock)
639 m_configFile.servers[m_serverIndex].locations[m_locationIndex].maxBodySize = size;
640}

815 if (returnCode < StatusOK || returnCode > StatusNonSupportedVersion)
816 throw std::runtime_error(ERR_INVALID_RETURN_CODE);

1055void ConfigFileParser::removeEnclosingDoubleQuotes(std::string& str)
1056{
1057 size_t leadingDoubleQuotes = 0;
1058 size_t trailingDoubleQuotes = 0;
1059
1060 while (leadingDoubleQuotes < str.length() && str[leadingDoubleQuotes] == '"') {
1061 leadingDoubleQuotes++;
1062 }
1063
1064 size_t trailingIndex = str.length();
1065 while (trailingIndex > leadingDoubleQuotes && str[trailingIndex - 1] == '"') {
1066 trailingDoubleQuotes++;
1067 trailingIndex--;
1068 }
1069
1070 if (leadingDoubleQuotes > 1 || trailingDoubleQuotes > 1)
1071 throw std::runtime_error(ERR_TOO_MANY_DOUBLE_QUOTES);
1072
1073 str = str.substr(leadingDoubleQuotes, trailingIndex - leadingDoubleQuotes);
1074}
