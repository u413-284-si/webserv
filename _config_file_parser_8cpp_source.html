50const ConfigFile& ConfigFileParser::parseConfigFile(const std::string& configFilePath)
51{
52 std::ifstream fileStream(configFilePath.c_str());
53 std::stringstream bufferStream;
54
55 if (!fileStream.is_open())
56 throw std::runtime_error(ERR_CONFIG_FILE_OPEN_FAILED);
57 if (fileStream.peek() == std::ifstream::traits_type::eof())
58 throw std::runtime_error(ERR_CONFIG_FILE_EMPTY);
59
60 bufferStream << fileStream.rdbuf();
61 m_configFileContent = bufferStream.str();
62 fileStream.close();
63
64 if (isBracketOpen())
65 throw std::runtime_error(ERR_OPEN_BRACKET_IN_CONFIG_FILE);
66
67 if (!isValidBlockBegin(HttpBlock))
68 throw std::runtime_error(ERR_MISSING_HTTP_BLOCK);
69
70 skipBlockBegin(HttpBlock);
71
72 while (m_configFileContent[m_configFileIndex] != '}') {
73 if (isValidBlockBegin(ServerBlock))
74 readServerBlock();
75 else if (std::isspace(m_configFileContent[m_configFileIndex]) == 0)
76 throw std::runtime_error(ERR_INVALID_DIRECTIVE);
77 m_configFileIndex++;
78 }
79
80 if (m_serverBlocksConfig.empty())
81 throw std::runtime_error(ERR_MISSING_SERVER_BLOCK);
82
83 for (std::vector<ServerBlockConfig>::const_iterator serverIt = m_serverBlocksConfig.begin();
84 serverIt != m_serverBlocksConfig.end(); serverIt++) {
85 processServerContent(*serverIt);
86 m_serverIndex++;
87 }
88
89 if (isLocationDuplicate())
90 throw std::runtime_error(ERR_DUPLICATE_LOCATION);
91
92 return m_configFile;
93}

120{

527void ConfigFileParser::readListen(const std::string& value)
528{
529 const size_t colonIndex = value.find(':');
530 const size_t endIndex = value.length();
531 const size_t dot = value.find('.');
532
533 if (colonIndex != std::string::npos) {
534 std::string ipAddress = value.substr(0, colonIndex);
535 if (!webutils::isIpAddressValid(ipAddress))
536 throw std::runtime_error(ERR_INVALID_IP_ADDRESS);
537
538 if (ipAddress == "localhost")
539 m_configFile.servers[m_serverIndex].host = "127.0.0.1";
540 else
541 m_configFile.servers[m_serverIndex].host = ipAddress;
542
543 std::string port = value.substr(colonIndex + 1, endIndex - colonIndex - 1);
544 if (port.find_first_of(s_whitespace) != std::string::npos)
545 throw std::runtime_error(ERR_INVALID_LISTEN_PARAMETERS);
546 if (!webutils::isPortValid(port))
547 throw std::runtime_error(ERR_INVALID_PORT);
548 m_configFile.servers[m_serverIndex].port = port;
549 return;
550 }
551
552 if (dot == std::string::npos) {
553 std::string hostOrPort = value.substr(0, endIndex);
554 if (hostOrPort.find_first_of(s_whitespace) != std::string::npos)
555 throw std::runtime_error(ERR_INVALID_LISTEN_PARAMETERS);
556 if (hostOrPort.empty())
557 throw std::runtime_error(ERR_EMPTY_LISTEN_VALUE);
558 if (hostOrPort == "localhost")
559 m_configFile.servers[m_serverIndex].host = "127.0.0.1";
560 else {
561 if (!webutils::isPortValid(hostOrPort))
562 throw std::runtime_error(ERR_INVALID_PORT);
563 m_configFile.servers[m_serverIndex].port = hostOrPort;
564 }
565 return;
566 }
567
568 std::string ipAddress = value.substr(0, endIndex);
569 if (ipAddress.find_first_of(s_whitespace) != std::string::npos)
570 throw std::runtime_error(ERR_INVALID_LISTEN_PARAMETERS);
571 if (!webutils::isIpAddressValid(ipAddress))
572 throw std::runtime_error(ERR_INVALID_IP_ADDRESS);
573
574 m_configFile.servers[m_serverIndex].host = ipAddress;
575}

600void ConfigFileParser::readMaxBodySize(const Block& block, const std::string& maxBodySize)
601{
602 if (maxBodySize.find_first_of(s_whitespace) != std::string::npos)
603 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_PARAMETERS);
604
605 const size_t lastNumberIndex = maxBodySize.find_last_of(s_number);
606 if (lastNumberIndex == std::string::npos)
607 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_VALUE);
608
609 const std::string number = maxBodySize.substr(0, lastNumberIndex + 1);
610 if (number.find_first_not_of(s_number) != std::string::npos)
611 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_VALUE);
612
613 errno = 0;
614 size_t size = std::strtoul(number.c_str(), NULL, constants::g_decimalBase);
615 if (errno == ERANGE)
616 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_NUMBER_OVERFLOW);
617
618 if (lastNumberIndex != maxBodySize.size() - 1) {
619 const std::string letter = maxBodySize.substr(lastNumberIndex + 1);
620 if (letter.length() != 1)
621 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_UNIT);
622
623 size_t unit = 1;
624 switch (letter.at(0)) {
625 case 'k':
626 case 'K':
627 unit = constants::g_oneKilobyte;
628 break;
629 case 'm':
630 case 'M':
631 unit = constants::g_oneMegabyte;
632 break;
633 case 'g':
634 case 'G':
635 unit = constants::g_oneGigabyte;
636 break;
637 default:
638 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_UNIT);
639 }
640
641 if (size > std::numeric_limits<size_t>::max() / unit)
642 throw std::runtime_error(ERR_INVALID_MAX_BODY_SIZE_UNIT_OVERFLOW);
643
644 size *= unit;
645 }
646
647 if (block == ServerBlock)
648 m_configFile.servers[m_serverIndex].maxBodySize = size;
649 else if (block == LocationBlock)
650 m_configFile.servers[m_serverIndex].locations[m_locationIndex].maxBodySize = size;
651}

1062void ConfigFileParser::removeEnclosingDoubleQuotes(std::string& str)
1063{
1064 size_t leadingDoubleQuotes = 0;
1065 size_t trailingDoubleQuotes = 0;
1066
1067 while (leadingDoubleQuotes < str.length() && str[leadingDoubleQuotes] == '"') {
1068 leadingDoubleQuotes++;
1069 }
1070
1071 size_t trailingIndex = str.length();
1072 while (trailingIndex > leadingDoubleQuotes && str[trailingIndex - 1] == '"') {
1073 trailingDoubleQuotes++;
1074 trailingIndex--;
1075 }
1076
1077 if (leadingDoubleQuotes > 1 || trailingDoubleQuotes > 1)
1078 throw std::runtime_error(ERR_TOO_MANY_DOUBLE_QUOTES);
1079
1080 str = str.substr(leadingDoubleQuotes, trailingIndex - leadingDoubleQuotes);
1081}
