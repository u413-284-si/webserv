186void CGIHandler::execute(
187 int epollFd, const std::map<int, Connection>& connections, const std::map<int, Connection*>& cgiConnections)
188{
189 if (m_processOps.forkProcess(m_cgiPid) == -1) {
190 closePipes();
191 m_request.httpStatus = StatusInternalServerError;
192 return;
193 }
194
195 if (m_cgiPid == 0) {
196 if (!registerChildSignals()) {
197 closeAllFds(epollFd, connections, cgiConnections);
198 std::exit(EXIT_FAILURE);
199 }
200
201 // Replace child stdin with read end of input pipe
202 if (m_request.method == MethodPost && m_processOps.dup2Process(m_pipeIn[0], STDIN_FILENO) == -1) {
203 closeAllFds(epollFd, connections, cgiConnections);
204 std::exit(EXIT_FAILURE);
205 }
206 // Replace child stdout with write end of output pipe
207 if (m_processOps.dup2Process(m_pipeOut[1], STDOUT_FILENO) == -1) {
208 closeAllFds(epollFd, connections, cgiConnections);
209 std::exit(EXIT_FAILURE);
210 }
211 closeAllFds(epollFd, connections, cgiConnections);
212
213 std::string workingDir = m_cgiScriptPath;
214 if (m_processOps.chdirProcess(workingDir.c_str()) == -1)
215 std::exit(EXIT_FAILURE);
216
217 if (m_processOps.execProcess(m_argvp[0], m_argvp.data(), m_envp.data()) == -1)
218 std::exit(EXIT_FAILURE);
219 }
220
221 // Close read end of input pipe in parent process
222 webutils::closeFd(m_pipeIn[0]);
223 // Close write end of output pipe in parent process
224 webutils::closeFd(m_pipeOut[1]);
225}
