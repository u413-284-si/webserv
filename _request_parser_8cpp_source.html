151std::string RequestParser::parseMethod(const std::string& requestLine, HTTPRequest& request)
152{
153 const int getLength = 3; // GET -> 3 chars
154 const int postLength = 4; // POST -> 4 chars
155 const int deleteLength = 6; // DELETE -> 6 chars
156 int index = 0;
157
158 if (requestLine.compare(0, getLength, "GET") == 0) {
159 request.method = MethodGet;
160 index = getLength;
161 } else if (requestLine.compare(0, postLength, "POST") == 0) {
162 request.method = MethodPost;
163 index = postLength;
164 } else if (requestLine.compare(0, deleteLength, "DELETE") == 0) {
165 request.method = MethodDelete;
166 index = deleteLength;

169 throw std::runtime_error(ERR_METHOD_NOT_IMPLEMENTED);
170 }
171 return (requestLine.substr(index));

191std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
192{
193 int index = 0;
194 if (requestLine.at(index) != '/') {
195 request.httpStatus = StatusBadRequest;
196 request.shallCloseConnection = true;
197 throw std::runtime_error(ERR_URI_MISS_SLASH);
198 }
199
200 // Check URI string for invalid chars
201 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
202 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos) {
203 request.httpStatus = StatusBadRequest;
204 request.shallCloseConnection = true;
205 throw std::runtime_error(ERR_URI_INVALID_CHAR);
206 }
207
208 request.uri.path.push_back(requestLine[index]);
209 while (requestLine.at(++index) != 0) {
210 if (requestLine.at(index) == ' ')
211 break;
212 if (requestLine.at(index) == '?')
213 parseUriQuery(requestLine, index, request);
214 else if (requestLine.at(index) == '#')
215 parseUriFragment(requestLine, index, request);
216 else
217 request.uri.path.push_back(requestLine.at(index));
218 }
219 return (requestLine.substr(index));
220}

302std::string RequestParser::parseVersion(const std::string& requestLine, HTTPRequest& request)
303{
304 const int versionPrefixLength = 5; // HTTP/ -> 5 chars
305
306 if (requestLine.substr(0, versionPrefixLength) != "HTTP/") {
307 request.httpStatus = StatusBadRequest;
308 request.shallCloseConnection = true;
309 throw std::runtime_error(ERR_INVALID_VERSION_FORMAT);
310 }

312 if (isdigit(requestLine[index]) == 0) {
313 request.httpStatus = StatusBadRequest;

315 throw std::runtime_error(ERR_INVALID_VERSION_MAJOR);

334 request.httpStatus = StatusNonSupportedVersion;

336 }

361void RequestParser::parseHeaders(HTTPRequest& request)
362{
363 std::string headerLine;
364
365 // The end of the headers section is marked by an empty line (\r\n\r\n).
366 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
367 if (headerLine[0] == ' ' || headerLine[0] == '\t') {
368 request.httpStatus = StatusBadRequest;
369 request.shallCloseConnection = true;
370 throw std::runtime_error(ERR_OBSOLETE_LINE_FOLDING);
371 }
372 std::string headerName;
373 std::string headerValue;
374 const std::size_t delimiterPos = headerLine.find_first_of(':');
375 if (delimiterPos != std::string::npos) {
376 headerName = headerLine.substr(0, delimiterPos);
377 validateHeaderName(headerName, request);
378 webutils::lowercase(headerName);
379 headerValue = headerLine.substr(delimiterPos + 1);
380 if (headerValue[headerValue.size() - 1] == '\r')
381 headerValue.erase(headerValue.size() - 1);
382 headerValue = webutils::trimLeadingWhitespaces(headerValue);
383 webutils::trimTrailingWhiteSpaces(headerValue);
384 validateContentLength(headerName, headerValue, request);
385 validateNoMultipleHostHeaders(headerName, request);
386 request.headers[headerName] = headerValue;
387 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
388 }
389 }
390 validateHostHeader(request);
391 validateTransferEncoding(request);
392 validateMethodWithBody(request);
393}

419void RequestParser::parseChunkedBody(HTTPRequest& request)
420{
421 size_t length = 0;
422 std::string strChunkSize;
423 std::string chunkData;
424 size_t numChunkSize = 0;
425
426 do {
427 std::getline(m_requestStream, strChunkSize);
428 if (strChunkSize[strChunkSize.size() - 1] == '\r')
429 strChunkSize.erase(strChunkSize.size() - 1);
430 else {
431 request.httpStatus = StatusBadRequest;
432 request.shallCloseConnection = true;
433 throw std::runtime_error(ERR_MISS_CRLF);
434 }
435 numChunkSize = convertHex(strChunkSize);
436 std::getline(m_requestStream, chunkData);
437 if (chunkData[chunkData.size() - 1] == '\r')
438 chunkData.erase(chunkData.size() - 1);
439 else {
440 request.httpStatus = StatusBadRequest;
441 request.shallCloseConnection = true;
442 throw std::runtime_error(ERR_MISS_CRLF);
443 }
444 if (chunkData.size() != numChunkSize) {
445 request.httpStatus = StatusBadRequest;
446 request.shallCloseConnection = true;
447 throw std::runtime_error(ERR_CHUNK_SIZE);
448 }
449 request.body += chunkData;
450 length += numChunkSize;
451 } while (numChunkSize > 0);
452 request.headers["content-length"] = webutils::toString(length);
453}

550{

671void RequestParser::validateHostHeader(HTTPRequest& request)
672{
673 LOG_DEBUG << "Validating Host header...";
674
675 std::map<std::string, std::string>::const_iterator iter = request.headers.find("host");
676 if (iter == request.headers.end()) {
677 request.httpStatus = StatusBadRequest;
678 request.shallCloseConnection = true;
679 throw std::runtime_error(ERR_MISSING_HOST_HEADER);
680 }
681
682 if (iter->second.empty()) {
683 request.httpStatus = StatusBadRequest;
684 request.shallCloseConnection = true;
685 throw std::runtime_error(ERR_EMPTY_HOST_VALUE);
686 }
687
688 if (iter->second.find(':') != std::string::npos) {
689 if (!webutils::isIpAddressValid(iter->second.substr(0, iter->second.find(':'))) ||
690 !webutils::isPortValid(iter->second.substr(iter->second.find(':') + 1))) {
691 request.httpStatus = StatusBadRequest;
692 request.shallCloseConnection = true;
693 throw std::runtime_error(ERR_INVALID_HOST_IP_WITH_PORT);
694 }
695 } else if (iter->second.find_first_not_of("0123456789.") == std::string::npos) {
696 if (!webutils::isIpAddressValid(iter->second.substr(0, iter->second.find(':')))) {
697 request.httpStatus = StatusBadRequest;
698 request.shallCloseConnection = true;
699 throw std::runtime_error(ERR_INVALID_HOST_IP);
700 }
701 } else {
702 if (!isValidHostname(iter->second)) {
703 request.httpStatus = StatusBadRequest;
704 request.shallCloseConnection = true;
705 throw std::runtime_error(ERR_INVALID_HOSTNAME);
706 }
707 }
708
709 LOG_DEBUG << "Valid host header: " << iter->second;
710}
