358  while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
359  if (headerLine[0] == ' ' || headerLine[0] == '\t') {
360  request.httpStatus = StatusBadRequest;

368  validateHeaderName(headerName, request);
369  webutils::lowercase(headerName);
370  headerValue = headerLine.substr(delimiterPos + 1);
371  if (headerValue[headerValue.size() - 1] == '\r')
372  headerValue.erase(headerValue.size() - 1);
373  headerValue = webutils::trimLeadingWhitespaces(headerValue);
374  webutils::trimTrailingWhiteSpaces(headerValue);

379  }
