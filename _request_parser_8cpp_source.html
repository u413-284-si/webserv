204std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
205{
206 int index = 0;
207 if (requestLine.at(index) != '/') {
208 request.httpStatus = StatusBadRequest;
209 request.shallCloseConnection = true;
210 throw std::runtime_error(ERR_URI_MISS_SLASH);
211 }
212
213 // Check URI string for invalid chars
214 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
215 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos) {
216 request.httpStatus = StatusBadRequest;
217 request.shallCloseConnection = true;
218 throw std::runtime_error(ERR_URI_INVALID_CHAR);
219 }
220
221 request.uri.path.push_back(requestLine[index]);
222 while (requestLine.at(++index) != 0) {
223 if (requestLine.at(index) == ' ')
224 break;
225 if (requestLine.at(index) == '?')
226 parseUriQuery(requestLine, index, request);
227 else if (requestLine.at(index) == '#')
228 parseUriFragment(requestLine, index, request);
229 else
230 request.uri.path.push_back(requestLine.at(index));
231 }
232 request.uri.path = decodePercentEncoding(request.uri.path, request);
233 request.uri.fragment = decodePercentEncoding(request.uri.fragment, request);
234 request.uri.query = decodePercentEncoding(request.uri.query, request);
235
236 request.uri.path = removeDotSegments(request.uri.path, request);
237 return (requestLine.substr(index));

379void RequestParser::parseHeaders(HTTPRequest& request)
380{
381 std::string headerLine;
382
383 // The end of the headers section is marked by an empty line (\r\n\r\n).
384 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
385 if (headerLine[0] == ' ' || headerLine[0] == '\t') {
386 request.httpStatus = StatusBadRequest;
387 request.shallCloseConnection = true;
388 throw std::runtime_error(ERR_OBSOLETE_LINE_FOLDING);
389 }
390 std::string headerName;
391 std::string headerValue;
392 const std::size_t delimiterPos = headerLine.find_first_of(':');
393 if (delimiterPos != std::string::npos) {
394 headerName = headerLine.substr(0, delimiterPos);
395 validateHeaderName(headerName, request);
396 webutils::lowercase(headerName);
397 headerValue = headerLine.substr(delimiterPos + 1);
398 if (headerValue[headerValue.size() - 1] == '\r')
399 headerValue.erase(headerValue.size() - 1);
400 headerValue = webutils::trimLeadingWhitespaces(headerValue);
401 webutils::trimTrailingWhiteSpaces(headerValue);
402 validateContentLength(headerName, headerValue, request);
403 validateNoMultipleHostHeaders(headerName, request);
404 request.headers[headerName] = headerValue;
405 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
406 }
407 }
408 validateHostHeader(request);
409 validateTransferEncoding(request);
410 validateMethodWithBody(request);
411}

427void RequestParser::parseChunkedBody(std::string& bodyBuffer, HTTPRequest& request)
428{
429 LOG_DEBUG << "Parsing chunked body...";
430
431 while (!bodyBuffer.empty()) {
432
433 // Step 1: Parse chunk size
434 if (request.chunkSize == -1) { // Not currently parsing a chunk
435 const size_t newlinePos = bodyBuffer.find("\r\n");
436 if (newlinePos == std::string::npos) // Incomplete chunk size indication, wait for more data
437 return;
438
439 std::string strChunkSize = bodyBuffer.substr(0, newlinePos);
440 bodyBuffer.erase(0, newlinePos + 2); // Remove the parsed chunk size and \r\n
441
442 request.chunkSize = convertHex(strChunkSize);
443 if (request.chunkSize > s_maxChunkSize) {
444 request.httpStatus = StatusRequestEntityTooLarge;
445 request.shallCloseConnection = true;
446 throw std::runtime_error(ERR_TOO_LARGE_CHUNKSIZE);
447 }
448
449 if (request.chunkSize == 0) { // Zero chunk size indicates the end of the body
450 if (bodyBuffer.size() != 2 || bodyBuffer.at(0) != '\r'
451 || bodyBuffer.at(1) != '\n') { // Check for final CRLF
452 request.httpStatus = StatusBadRequest;
453 request.shallCloseConnection = true;
454 throw std::runtime_error(ERR_MISS_CRLF);
455 }
456 request.isCompleteBody = true;
457 request.headers["content-length"] = webutils::toString(request.body.size());
458 LOG_DEBUG << "Successfully parsed chunked body";
459 return;
460 }
461 }
462
463 // Step 2: Parse chunk data
464 const size_t remainingData = bodyBuffer.size();
465 const size_t requiredData = request.chunkSize + 2; // Chunk data + \r\n
466
467 if (remainingData < requiredData) // Incomplete chunk data, wait for more data
468 return;
469
470 if (bodyBuffer.at(request.chunkSize) != '\r' || bodyBuffer.at(request.chunkSize + 1) != '\n') {
471 request.httpStatus = StatusBadRequest;
472 request.shallCloseConnection = true;
473 throw std::runtime_error(ERR_MISS_CRLF);
474 }
475
476 request.body.append(bodyBuffer, 0, request.chunkSize);
477 bodyBuffer.erase(0, requiredData);
478
479 // Reset for next chunk
480 request.chunkSize = -1;
481 }
482}

495void RequestParser::decodeMultipartFormdata(HTTPRequest& request)
496{
497 size_t currentBoundaryEndPos = checkForString("--" + request.boundary, 0, request.body);
498 bool isFirstFileUpload = true;
499 std::string extractedBody;
500 const std::string filename = "filename=\"";
501
502 while (true) {
503 currentBoundaryEndPos += request.boundary.size() + 2;
504 size_t nextBoundaryPos = checkForString("--" + request.boundary, currentBoundaryEndPos, request.body);
505
506 // Find form header section and lower case it
507 size_t contentStartPos = checkForString("\r\n\r\n", currentBoundaryEndPos, request.body);
508 std::string loweredFormHeader
509 = request.body.substr(currentBoundaryEndPos, contentStartPos - currentBoundaryEndPos);
510 webutils::lowercase(loweredFormHeader);
511
512 // Check required headers
513 const size_t dispositionPos = checkForString("content-disposition:", 0, loweredFormHeader);
514 size_t filenameStartPos = loweredFormHeader.find(filename, dispositionPos);
515 if (filenameStartPos != std::string::npos) {
516 filenameStartPos += filename.size();
517 checkForString("content-type", filenameStartPos, loweredFormHeader);
518
519 // Process file upload data
520 if (isFirstFileUpload) {
521 const size_t filenameEndPos = checkForString("\"", filenameStartPos, loweredFormHeader);
522 request.targetResource
523 += request.body.substr(currentBoundaryEndPos + filenameStartPos, filenameEndPos - filenameStartPos);
524 LOG_DEBUG << "New target resource: " << request.targetResource;
525
526 contentStartPos += 4; // Skip the CRLFCRLF
527 const size_t contentEndPos = nextBoundaryPos - 2; // Remove the CRLF at the end
528 extractedBody = request.body.substr(contentStartPos, contentEndPos - contentStartPos);
529
530 isFirstFileUpload = false;
531 } else
532 throw std::runtime_error(ERR_MULTIPLE_UPLOADS);
533 }
534
535 // Check for two dashes after boundary indicating the end boundary
536 if (request.body.at(nextBoundaryPos + request.boundary.size() + 2) == '-'
537 && request.body.at(nextBoundaryPos + request.boundary.size() + 3) == '-')
538 break;
539
540 currentBoundaryEndPos = nextBoundaryPos;
541 }
542
543 if (extractedBody.empty())
544 throw std::runtime_error(ERR_BAD_MULTIPART_FORMDATA);
545
546 request.body = extractedBody;
547}
