180std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
181{
182 int index = 0;
183 if (requestLine.at(index) != '/') {
184 request.httpStatus = StatusBadRequest;
185 request.shallCloseConnection = true;
186 throw std::runtime_error(ERR_URI_MISS_SLASH);
187 }
188
189 // Check URI string for invalid chars
190 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
191 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos) {
192 request.httpStatus = StatusBadRequest;
193 request.shallCloseConnection = true;
194 throw std::runtime_error(ERR_URI_INVALID_CHAR);

199 if (requestLine.at(index) == ' ')

208 request.uri.path = decodePercentEncoding(request.uri.path, request);
209 request.uri.fragment = decodePercentEncoding(request.uri.fragment, request);
210 request.uri.query = decodePercentEncoding(request.uri.query, request);

356{
357 std::string headerLine;
358
359 // The end of the headers section is marked by an empty line (\r\n\r\n).
360 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
361 if (headerLine[0] == ' ' || headerLine[0] == '\t') {
362 request.httpStatus = StatusBadRequest;
363 request.shallCloseConnection = true;
364 throw std::runtime_error(ERR_OBSOLETE_LINE_FOLDING);
365 }
366 const std::size_t delimiterPos = headerLine.find_first_of(':');
367 if (delimiterPos != std::string::npos) {
368 const std::string headerName = webutils::lowercase(headerLine.substr(0, delimiterPos));
369 validateHeaderName(headerName, request);
370 std::string headerValue = headerLine.substr(delimiterPos + 1);
371 if (headerValue[headerValue.size() - 1] == '\r')
372 headerValue.erase(headerValue.size() - 1);
373 headerValue = webutils::trimLeadingWhitespaces(headerValue);
374 webutils::trimTrailingWhiteSpaces(headerValue);
375 validateContentLength(headerName, headerValue, request);
376 validateNoMultipleHostHeaders(headerName, request);
377 request.headers[headerName] = headerValue;
378 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
379 }
380 }
381 validateHostHeader(request);
382 validateTransferEncoding(request);
383 validateMethodWithBody(request);
384 validateConnectionHeader(request);

401void RequestParser::parseChunkedBody(std::string& bodyBuffer, HTTPRequest& request)
402{
403 LOG_DEBUG << "Parsing chunked body...";
404
405 while (!bodyBuffer.empty()) {
406
407 // Step 1: Parse chunk size
408 if (request.chunkSize == -1) { // Not currently parsing a chunk
409 const size_t newlinePos = bodyBuffer.find("\r\n");
410 if (newlinePos == std::string::npos) // Incomplete chunk size indication, wait for more data
411 return;
412
413 std::string strChunkSize = bodyBuffer.substr(0, newlinePos);
414 bodyBuffer.erase(0, newlinePos + 2); // Remove the parsed chunk size and \r\n
415
416 request.chunkSize = convertHex(strChunkSize, request);
417 if (request.chunkSize > s_maxChunkSize) {
418 request.httpStatus = StatusRequestEntityTooLarge;
419 request.shallCloseConnection = true;
420 throw std::runtime_error(ERR_TOO_LARGE_CHUNKSIZE);
421 }
422
423 if (request.chunkSize == 0) { // Zero chunk size indicates the end of the body
424 if (bodyBuffer.size() != 2 || bodyBuffer.at(0) != '\r'
425 || bodyBuffer.at(1) != '\n') { // Check for final CRLF
426 request.httpStatus = StatusBadRequest;
427 request.shallCloseConnection = true;
428 throw std::runtime_error(ERR_MISS_CRLF);
429 }
430 request.isCompleteBody = true;
431 request.headers["content-length"] = webutils::toString(request.body.size());
432 LOG_DEBUG << "Successfully parsed chunked body";
433 return;
434 }
435 }
436
437 // Step 2: Parse chunk data
438 const size_t remainingData = bodyBuffer.size();
439 const size_t requiredData = request.chunkSize + 2; // Chunk data + \r\n
440
441 if (remainingData < requiredData) // Incomplete chunk data, wait for more data
442 return;
443
444 if (bodyBuffer.at(request.chunkSize) != '\r' || bodyBuffer.at(request.chunkSize + 1) != '\n') {
445 request.httpStatus = StatusBadRequest;
446 request.shallCloseConnection = true;
447 throw std::runtime_error(ERR_MISS_CRLF);
448 }
449
450 request.body.append(bodyBuffer, 0, request.chunkSize);
451 bodyBuffer.erase(0, requiredData);
452
453 // Reset for next chunk
454 request.chunkSize = -1;
455 }
456}

469void RequestParser::decodeMultipartFormdata(HTTPRequest& request)
470{
471 size_t currentBoundaryEndPos = checkForString("--" + request.boundary, 0, request.body);
472 bool isFirstFileUpload = true;
473 std::string extractedBody;
474 const std::string filename = "filename=\"";
475
476 while (true) {
477 currentBoundaryEndPos += request.boundary.size() + 2;
478 size_t nextBoundaryPos = checkForString("--" + request.boundary, currentBoundaryEndPos, request.body);
479
480 // Find form header section and lower case it
481 size_t contentStartPos = checkForString("\r\n\r\n", currentBoundaryEndPos, request.body);
482 const std::string loweredFormHeader
483 = webutils::lowercase(request.body.substr(currentBoundaryEndPos, contentStartPos - currentBoundaryEndPos));
484
485 // Check required headers
486 const size_t dispositionPos = checkForString("content-disposition:", 0, loweredFormHeader);
487 size_t filenameStartPos = loweredFormHeader.find(filename, dispositionPos);
488 if (filenameStartPos != std::string::npos) {
489 filenameStartPos += filename.size();
490 checkForString("content-type", filenameStartPos, loweredFormHeader);
491
492 // Process file upload data
493 if (isFirstFileUpload) {
494 const size_t filenameEndPos = checkForString("\"", filenameStartPos, loweredFormHeader);
495 request.targetResource
496 += request.body.substr(currentBoundaryEndPos + filenameStartPos, filenameEndPos - filenameStartPos);
497 LOG_DEBUG << "New target resource: " << request.targetResource;
498
499 contentStartPos += 4; // Skip the CRLFCRLF
500 const size_t contentEndPos = nextBoundaryPos - 2; // Remove the CRLF at the end
501 extractedBody = request.body.substr(contentStartPos, contentEndPos - contentStartPos);
502
503 isFirstFileUpload = false;
504 } else
505 throw std::runtime_error(ERR_MULTIPLE_UPLOADS);
506 }
507
508 // Check for two dashes after boundary indicating the end boundary
509 if (request.body.at(nextBoundaryPos + request.boundary.size() + 2) == '-'
510 && request.body.at(nextBoundaryPos + request.boundary.size() + 3) == '-')
511 break;
512
513 currentBoundaryEndPos = nextBoundaryPos;
514 }
515
516 if (extractedBody.empty())
517 throw std::runtime_error(ERR_BAD_MULTIPART_FORMDATA);
518
519 request.body = extractedBody;
520}
