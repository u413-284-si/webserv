193std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
194{
195 int index = 0;
196 if (requestLine.at(index) != '/') {
197 request.httpStatus = StatusBadRequest;
198 request.shallCloseConnection = true;
199 throw std::runtime_error(ERR_URI_MISS_SLASH);
200 }
201
202 // Check URI string for invalid chars
203 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
204 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos) {
205 request.httpStatus = StatusBadRequest;
206 request.shallCloseConnection = true;
207 throw std::runtime_error(ERR_URI_INVALID_CHAR);
208 }
209
210 request.uri.path.push_back(requestLine[index]);
211 while (requestLine.at(++index) != 0) {
212 if (requestLine.at(index) == ' ')
213 break;
214 if (requestLine.at(index) == '?')
215 parseUriQuery(requestLine, index, request);
216 else if (requestLine.at(index) == '#')
217 parseUriFragment(requestLine, index, request);
218 else
219 request.uri.path.push_back(requestLine.at(index));
220 }
221 request.uri.path = decodePercentEncoding(request.uri.path, request);
222 request.uri.fragment = decodePercentEncoding(request.uri.fragment, request);
223 request.uri.query = decodePercentEncoding(request.uri.query, request);
224
225 request.uri.path = removeDotSegments(request.uri.path, request);
226 return (requestLine.substr(index));
227}

309std::string RequestParser::parseVersion(const std::string& requestLine, HTTPRequest& request)
310{
311 const int versionPrefixLength = 5; // HTTP/ -> 5 chars
312
313 if (requestLine.substr(0, versionPrefixLength) != "HTTP/") {
314 request.httpStatus = StatusBadRequest;
315 request.shallCloseConnection = true;
316 throw std::runtime_error(ERR_INVALID_VERSION_FORMAT);
317 }
318 int index = versionPrefixLength;
319 if (isdigit(requestLine[index]) == 0) {
320 request.httpStatus = StatusBadRequest;
321 request.shallCloseConnection = true;
322 throw std::runtime_error(ERR_INVALID_VERSION_MAJOR);
323 }
324 if (requestLine[index] != '1') {
325 request.httpStatus = StatusNonSupportedVersion;
326 throw std::runtime_error(ERR_NONSUPPORTED_VERSION);
327 }
328 request.version.push_back(requestLine[index]);
329 if (requestLine[++index] != '.') {
330 request.httpStatus = StatusBadRequest;
331 request.shallCloseConnection = true;
332 throw std::runtime_error(ERR_INVALID_VERSION_DELIM);
333 }
334 request.version.push_back(requestLine[index]);
335 if (isdigit(requestLine[++index]) == 0) {
336 request.httpStatus = StatusBadRequest;
337 request.shallCloseConnection = true;
338 throw std::runtime_error(ERR_INVALID_VERSION_MINOR);
339 }
340 if (requestLine[index] != '1' && requestLine[index] != '0') {
341 request.httpStatus = StatusNonSupportedVersion;
342 throw std::runtime_error(ERR_NONSUPPORTED_VERSION);
343 }
344 request.version.push_back(requestLine[index]);

368void RequestParser::parseHeaders(HTTPRequest& request)
369{
370 std::string headerLine;
371
372 // The end of the headers section is marked by an empty line (\r\n\r\n).
373 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
374 if (headerLine[0] == ' ' || headerLine[0] == '\t') {
375 request.httpStatus = StatusBadRequest;
376 request.shallCloseConnection = true;
377 throw std::runtime_error(ERR_OBSOLETE_LINE_FOLDING);
378 }
379 std::string headerName;
380 std::string headerValue;
381 const std::size_t delimiterPos = headerLine.find_first_of(':');
382 if (delimiterPos != std::string::npos) {
383 headerName = headerLine.substr(0, delimiterPos);
384 validateHeaderName(headerName, request);
385 webutils::lowercase(headerName);
386 headerValue = headerLine.substr(delimiterPos + 1);
387 if (headerValue[headerValue.size() - 1] == '\r')
388 headerValue.erase(headerValue.size() - 1);
389 headerValue = webutils::trimLeadingWhitespaces(headerValue);
390 webutils::trimTrailingWhiteSpaces(headerValue);
391 validateContentLength(headerName, headerValue, request);
392 validateNoMultipleHostHeaders(headerName, request);
393 request.headers[headerName] = headerValue;
394 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
395 }
396 }
397 validateHostHeader(request);
398 validateTransferEncoding(request);
399 validateMethodWithBody(request);
400}

427void RequestParser::parseChunkedBody(HTTPRequest& request, std::vector<char>& buffer)
428{
429 LOG_DEBUG << "Parsing chunked body...";
430
431 size_t length = 0;
432 std::string strChunkSize;
433 size_t numChunkSize = 0;
434
435 do {
436 std::getline(m_requestStream, strChunkSize);
437 if (!strChunkSize.empty() && strChunkSize[strChunkSize.size() - 1] == '\r')
438 strChunkSize.erase(strChunkSize.size() - 1);
439 else {
440 request.httpStatus = StatusBadRequest;
441 request.shallCloseConnection = true;
442 throw std::runtime_error(ERR_MISS_CRLF);
443 }
444
445 numChunkSize = convertHex(strChunkSize);
446 if (numChunkSize > s_maxChunkSize) {
447 request.httpStatus = StatusRequestEntityTooLarge;
448 request.shallCloseConnection = true;
449 throw std::runtime_error(ERR_TOO_LARGE_CHUNKSIZE);
450 }
451
452 if (buffer.capacity() < numChunkSize + 2)
453 buffer.resize(numChunkSize + 2);
454
455 errno = 0;
456 m_requestStream.read(buffer.data(), static_cast<long>(numChunkSize + 2));
457 if (m_requestStream.gcount() != static_cast<std::streamsize>(numChunkSize + 2)) {
458 if (!m_requestStream.good() && !m_requestStream.eof()) {
459 request.httpStatus = StatusInternalServerError;
460 throw std::runtime_error("read(): " + std::string(strerror(errno)));
461 }
462 request.httpStatus = StatusBadRequest;
463 request.shallCloseConnection = true;
464 throw std::runtime_error(ERR_CHUNKSIZE_INCONSISTENT);
465 }
466
467 if (buffer.at(numChunkSize) == '\r' && buffer.at(numChunkSize + 1) == '\n') {
468 request.body.append(buffer.data(), numChunkSize);
469 } else {
470 request.httpStatus = StatusBadRequest;
471 request.shallCloseConnection = true;
472 throw std::runtime_error(ERR_MISS_CRLF);
473 }
474
475 length += numChunkSize;
476 } while (numChunkSize > 0);
477 request.headers["content-length"] = webutils::toString(length);
478
479 LOG_DEBUG << "Successfully parsed chunked body";
480}

1200 iter++;
1201
