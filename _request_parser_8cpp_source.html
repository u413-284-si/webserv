184std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
185{
186 int index = 0;
187 if (requestLine.at(index) != '/')
188 throw HTTPErrorException(StatusBadRequest, ERR_URI_MISS_SLASH);
189
190 // Check URI string for invalid chars
191 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
192 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos)
193 throw HTTPErrorException(StatusBadRequest, ERR_URI_INVALID_CHAR);
194
195 request.uri.path.push_back(requestLine[index]);
196 while (requestLine.at(++index) != 0) {
197 if (requestLine.at(index) == ' ')
198 break;
199 if (requestLine.at(index) == '?')
200 parseUriQuery(requestLine, index, request);
201 else if (requestLine.at(index) == '#')
202 parseUriFragment(requestLine, index, request);
203 else
204 request.uri.path.push_back(requestLine.at(index));
205 }
206 request.uri.path = decodePercentEncoding(request.uri.path);
207 request.uri.fragment = decodePercentEncoding(request.uri.fragment);
208 request.uri.query = decodePercentEncoding(request.uri.query);
209
210 request.uri.path = removeDotSegments(request.uri.path);
211 return (requestLine.substr(index));
212}

333void RequestParser::parseHeaders(HTTPRequest& request)
334{
335 std::string headerLine;
336
337 // The end of the headers section is marked by an empty line (\r\n\r\n).
338 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
339 if (headerLine[0] == ' ' || headerLine[0] == '\t')
340 throw HTTPErrorException(StatusBadRequest, ERR_OBSOLETE_LINE_FOLDING);
341
342 const std::size_t delimiterPos = headerLine.find_first_of(':');
343 if (delimiterPos != std::string::npos) {
344 const std::string headerName = webutils::lowercase(headerLine.substr(0, delimiterPos));
345 validateHeaderName(headerName);
346 std::string headerValue = headerLine.substr(delimiterPos + 1);
347 if (headerValue[headerValue.size() - 1] == '\r')
348 headerValue.erase(headerValue.size() - 1);
349 headerValue = webutils::trimWhiteSpaces(headerValue);
350 validateContentLength(headerName, headerValue, request);
351 validateNoMultipleHostHeaders(headerName, request);
352 request.headers[headerName] = headerValue;
353 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
354 }
355 }
356 validateHostHeader(request);
357 validateTransferEncoding(request);
358 validateMethodWithBody(request);
359 validateConnectionHeader(request);
360}

376void RequestParser::parseChunkedBody(std::string& bodyBuffer, HTTPRequest& request)
377{
378 LOG_DEBUG << "Parsing chunked body...";
379
380 while (!bodyBuffer.empty()) {
381
382 // Step 1: Parse chunk size
383 if (request.chunkSize == -1) { // Not currently parsing a chunk
384 const size_t newlinePos = bodyBuffer.find("\r\n");
385 if (newlinePos == std::string::npos) // Incomplete chunk size indication, wait for more data
386 return;
387
388 std::string strChunkSize = bodyBuffer.substr(0, newlinePos);
389 bodyBuffer.erase(0, newlinePos + 2); // Remove the parsed chunk size and \r\n
390
391 request.chunkSize = convertHex(strChunkSize);
392 if (request.chunkSize > s_maxChunkSize)
393 throw HTTPErrorException(StatusRequestEntityTooLarge, ERR_TOO_LARGE_CHUNKSIZE);
394
395 if (request.chunkSize == 0) { // Zero chunk size indicates the end of the body
396 if (bodyBuffer.size() != 2 || bodyBuffer.at(0) != '\r'
397 || bodyBuffer.at(1) != '\n') // Check for final CRLF
398 throw HTTPErrorException(StatusBadRequest, ERR_MISS_CRLF);
399 request.isCompleteBody = true;
400 request.headers["content-length"] = webutils::toString(request.body.size());
401 LOG_DEBUG << "Successfully parsed chunked body";
402 return;
403 }
404 }
405
406 // Step 2: Parse chunk data
407 const size_t remainingData = bodyBuffer.size();
408 const size_t requiredData = request.chunkSize + 2; // Chunk data + \r\n
409
410 if (remainingData < requiredData) // Incomplete chunk data, wait for more data
411 return;
412
413 if (bodyBuffer.at(request.chunkSize) != '\r' || bodyBuffer.at(request.chunkSize + 1) != '\n')
414 throw HTTPErrorException(StatusBadRequest, ERR_MISS_CRLF);
415
416 request.body.append(bodyBuffer, 0, request.chunkSize);
417 bodyBuffer.erase(0, requiredData);
418
419 // Reset for next chunk
420 request.chunkSize = -1;
421 }
422}

435void RequestParser::decodeMultipartFormdata(HTTPRequest& request)
436{
437 size_t currentBoundaryEndPos = checkForString("--" + request.boundary, 0, request.body);
438 bool isFirstFileUpload = true;
439 std::string extractedBody;
440 const std::string filename = "filename=\"";
441
442 while (true) {
443 currentBoundaryEndPos += request.boundary.size() + 2;
444 size_t nextBoundaryPos = checkForString("--" + request.boundary, currentBoundaryEndPos, request.body);
445
446 // Find form header section and lower case it
447 size_t contentStartPos = checkForString("\r\n\r\n", currentBoundaryEndPos, request.body);
448 const std::string loweredFormHeader
449 = webutils::lowercase(request.body.substr(currentBoundaryEndPos, contentStartPos - currentBoundaryEndPos));
450
451 // Check required headers
452 const size_t dispositionPos = checkForString("content-disposition:", 0, loweredFormHeader);
453 size_t filenameStartPos = loweredFormHeader.find(filename, dispositionPos);
454 if (filenameStartPos != std::string::npos) {
455 filenameStartPos += filename.size();
456 checkForString("content-type", filenameStartPos, loweredFormHeader);
457
458 // Process file upload data
459 if (isFirstFileUpload) {
460 const size_t filenameEndPos = checkForString("\"", filenameStartPos, loweredFormHeader);
461 request.targetResource
462 += request.body.substr(currentBoundaryEndPos + filenameStartPos, filenameEndPos - filenameStartPos);
463 LOG_DEBUG << "New target resource: " << request.targetResource;
464
465 contentStartPos += 4; // Skip the CRLFCRLF
466 const size_t contentEndPos = nextBoundaryPos - 2; // Remove the CRLF at the end
467 extractedBody = request.body.substr(contentStartPos, contentEndPos - contentStartPos);
468
469 isFirstFileUpload = false;
470 } else
471 throw HTTPErrorException(StatusBadRequest, ERR_MULTIPLE_UPLOADS);
472 }
473
474 // Check for two dashes after boundary indicating the end boundary
475 if (request.body.at(nextBoundaryPos + request.boundary.size() + 2) == '-'
476 && request.body.at(nextBoundaryPos + request.boundary.size() + 3) == '-')
477 break;
478
479 currentBoundaryEndPos = nextBoundaryPos;
480 }
481
482 if (extractedBody.empty())
483 throw HTTPErrorException(StatusBadRequest, ERR_BAD_MULTIPART_FORMDATA);
484
485 request.body = extractedBody;
486}
