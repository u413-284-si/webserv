178std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
179{
180 int index = 0;
181 if (requestLine.at(index) != '/') {
182 request.httpStatus = StatusBadRequest;
183 throw std::runtime_error(ERR_URI_MISS_SLASH);
184 }
185
186 // Check URI string for invalid chars
187 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
188 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos) {
189 request.httpStatus = StatusBadRequest;
190 throw std::runtime_error(ERR_URI_INVALID_CHAR);
191 }
192
193 request.uri.path.push_back(requestLine[index]);
194 while (requestLine.at(++index) != 0) {
195 if (requestLine.at(index) == ' ')
196 break;
197 if (requestLine.at(index) == '?')
198 parseUriQuery(requestLine, index, request);
199 else if (requestLine.at(index) == '#')
200 parseUriFragment(requestLine, index, request);
201 else
202 request.uri.path.push_back(requestLine.at(index));
203 }
204 request.uri.path = decodePercentEncoding(request.uri.path, request);
205 request.uri.fragment = decodePercentEncoding(request.uri.fragment, request);
206 request.uri.query = decodePercentEncoding(request.uri.query, request);
207
208 request.uri.path = removeDotSegments(request.uri.path, request);
209 return (requestLine.substr(index));

345void RequestParser::parseHeaders(HTTPRequest& request)
346{
347 std::string headerLine;
348
349 // The end of the headers section is marked by an empty line (\r\n\r\n).
350 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
351 if (headerLine[0] == ' ' || headerLine[0] == '\t') {
352 request.httpStatus = StatusBadRequest;
353 throw std::runtime_error(ERR_OBSOLETE_LINE_FOLDING);
354 }
355 const std::size_t delimiterPos = headerLine.find_first_of(':');
356 if (delimiterPos != std::string::npos) {
357 const std::string headerName = webutils::lowercase(headerLine.substr(0, delimiterPos));
358 validateHeaderName(headerName, request);
359 std::string headerValue = headerLine.substr(delimiterPos + 1);
360 if (headerValue[headerValue.size() - 1] == '\r')
361 headerValue.erase(headerValue.size() - 1);
362 headerValue = webutils::trimLeadingWhitespaces(headerValue);
363 webutils::trimTrailingWhiteSpaces(headerValue);
364 validateContentLength(headerName, headerValue, request);
365 validateNoMultipleHostHeaders(headerName, request);
366 request.headers[headerName] = headerValue;
367 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
368 }
369 }
370 validateHostHeader(request);
371 validateTransferEncoding(request);
372 validateMethodWithBody(request);
373 validateConnectionHeader(request);

390void RequestParser::parseChunkedBody(std::string& bodyBuffer, HTTPRequest& request)
391{
392 LOG_DEBUG << "Parsing chunked body...";
393
394 while (!bodyBuffer.empty()) {
395
396 // Step 1: Parse chunk size
397 if (request.chunkSize == -1) { // Not currently parsing a chunk
398 const size_t newlinePos = bodyBuffer.find("\r\n");
399 if (newlinePos == std::string::npos) // Incomplete chunk size indication, wait for more data
400 return;
401
402 std::string strChunkSize = bodyBuffer.substr(0, newlinePos);
403 bodyBuffer.erase(0, newlinePos + 2); // Remove the parsed chunk size and \r\n
404
405 request.chunkSize = convertHex(strChunkSize, request);
406 if (request.chunkSize > s_maxChunkSize) {
407 request.httpStatus = StatusRequestEntityTooLarge;
408 throw std::runtime_error(ERR_TOO_LARGE_CHUNKSIZE);
409 }
410
411 if (request.chunkSize == 0) { // Zero chunk size indicates the end of the body
412 if (bodyBuffer.size() != 2 || bodyBuffer.at(0) != '\r'
413 || bodyBuffer.at(1) != '\n') { // Check for final CRLF
414 request.httpStatus = StatusBadRequest;
415 throw std::runtime_error(ERR_MISS_CRLF);
416 }
417 request.isCompleteBody = true;
418 request.headers["content-length"] = webutils::toString(request.body.size());
419 LOG_DEBUG << "Successfully parsed chunked body";
420 return;
421 }
422 }
423
424 // Step 2: Parse chunk data
425 const size_t remainingData = bodyBuffer.size();
426 const size_t requiredData = request.chunkSize + 2; // Chunk data + \r\n
427
428 if (remainingData < requiredData) // Incomplete chunk data, wait for more data
429 return;
430
431 if (bodyBuffer.at(request.chunkSize) != '\r' || bodyBuffer.at(request.chunkSize + 1) != '\n') {
432 request.httpStatus = StatusBadRequest;
433 throw std::runtime_error(ERR_MISS_CRLF);
434 }
435
436 request.body.append(bodyBuffer, 0, request.chunkSize);
437 bodyBuffer.erase(0, requiredData);
438
439 // Reset for next chunk
440 request.chunkSize = -1;
441 }
442}

455void RequestParser::decodeMultipartFormdata(HTTPRequest& request)
456{
457 size_t currentBoundaryEndPos = checkForString("--" + request.boundary, 0, request.body);
458 bool isFirstFileUpload = true;
459 std::string extractedBody;
460 const std::string filename = "filename=\"";
461
462 while (true) {
463 currentBoundaryEndPos += request.boundary.size() + 2;
464 size_t nextBoundaryPos = checkForString("--" + request.boundary, currentBoundaryEndPos, request.body);
465
466 // Find form header section and lower case it
467 size_t contentStartPos = checkForString("\r\n\r\n", currentBoundaryEndPos, request.body);
468 const std::string loweredFormHeader
469 = webutils::lowercase(request.body.substr(currentBoundaryEndPos, contentStartPos - currentBoundaryEndPos));
470
471 // Check required headers
472 const size_t dispositionPos = checkForString("content-disposition:", 0, loweredFormHeader);
473 size_t filenameStartPos = loweredFormHeader.find(filename, dispositionPos);
474 if (filenameStartPos != std::string::npos) {
475 filenameStartPos += filename.size();
476 checkForString("content-type", filenameStartPos, loweredFormHeader);
477
478 // Process file upload data
479 if (isFirstFileUpload) {
480 const size_t filenameEndPos = checkForString("\"", filenameStartPos, loweredFormHeader);
481 request.targetResource
482 += request.body.substr(currentBoundaryEndPos + filenameStartPos, filenameEndPos - filenameStartPos);
483 LOG_DEBUG << "New target resource: " << request.targetResource;
484
485 contentStartPos += 4; // Skip the CRLFCRLF
486 const size_t contentEndPos = nextBoundaryPos - 2; // Remove the CRLF at the end
487 extractedBody = request.body.substr(contentStartPos, contentEndPos - contentStartPos);
488
489 isFirstFileUpload = false;
490 } else
491 throw std::runtime_error(ERR_MULTIPLE_UPLOADS);
492 }
493
494 // Check for two dashes after boundary indicating the end boundary
495 if (request.body.at(nextBoundaryPos + request.boundary.size() + 2) == '-'
496 && request.body.at(nextBoundaryPos + request.boundary.size() + 3) == '-')
497 break;
498
499 currentBoundaryEndPos = nextBoundaryPos;
500 }
501
502 if (extractedBody.empty())
503 throw std::runtime_error(ERR_BAD_MULTIPART_FORMDATA);
504
505 request.body = extractedBody;
506}

1124std::string RequestParser::decodePercentEncoding(const std::string& encoded, HTTPRequest& request)
1125{
1126 std::string decoded;
1127 std::string::const_iterator iter = encoded.begin();
1128
1129 while (iter != encoded.end()) {
1130 if (*iter == '%') {
1131 if (std::distance(iter, encoded.end()) < 3) {
1132 request.httpStatus = StatusBadRequest;
1133 throw std::runtime_error(ERR_PERCENT_INCOMPLETE);
1134 }
1135 if ((std::isxdigit(*(iter + 1)) == 0) || (std::isxdigit(*(iter + 2)) == 0)) {
1136 request.httpStatus = StatusBadRequest;
1137 throw std::runtime_error(ERR_PERCENT_INVALID_HEX);
1138 }

1140 unsigned int value = 0;
1141 std::istringstream(hex) >> std::hex >> value;
1142 if (value == 0) {
1143 request.httpStatus = StatusBadRequest;
1144 throw std::runtime_error(ERR_PERCENT_NONSUPPORTED_NUL);
