192std::string RequestParser::parseUri(const std::string& requestLine, HTTPRequest& request)
193{
194 int index = 0;
195 if (requestLine.at(index) != '/') {
196 request.httpStatus = StatusBadRequest;
197 request.shallCloseConnection = true;
198 throw std::runtime_error(ERR_URI_MISS_SLASH);
199 }
200
201 // Check URI string for invalid chars
202 const std::string::const_iterator delimiterPos = find(requestLine.begin(), requestLine.end(), ' ');
203 if (std::find_if(requestLine.begin(), delimiterPos, isNotValidURIChar) != delimiterPos) {
204 request.httpStatus = StatusBadRequest;
205 request.shallCloseConnection = true;
206 throw std::runtime_error(ERR_URI_INVALID_CHAR);
207 }
208
209 request.uri.path.push_back(requestLine[index]);
210 while (requestLine.at(++index) != 0) {
211 if (requestLine.at(index) == ' ')
212 break;
213 if (requestLine.at(index) == '?')
214 parseUriQuery(requestLine, index, request);
215 else if (requestLine.at(index) == '#')
216 parseUriFragment(requestLine, index, request);
217 else
218 request.uri.path.push_back(requestLine.at(index));
219 }
220 request.uri.path = decodePercentEncoding(request.uri.path, request);
221 request.uri.fragment = decodePercentEncoding(request.uri.fragment, request);
222 request.uri.query = decodePercentEncoding(request.uri.query, request);
223
224 request.uri.path = removeDotSegments(request.uri.path, request);
225 return (requestLine.substr(index));

308std::string RequestParser::parseVersion(const std::string& requestLine, HTTPRequest& request)
309{

315 throw std::runtime_error(ERR_INVALID_VERSION_FORMAT);
316 }

318 if (isdigit(requestLine[index]) == 0) {

367void RequestParser::parseHeaders(HTTPRequest& request)
368{
369 std::string headerLine;
370
371 // The end of the headers section is marked by an empty line (\r\n\r\n).
372 while (!std::getline(m_requestStream, headerLine).fail() && headerLine != "\r" && !headerLine.empty()) {
373 if (headerLine[0] == ' ' || headerLine[0] == '\t') {
374 request.httpStatus = StatusBadRequest;
375 request.shallCloseConnection = true;
376 throw std::runtime_error(ERR_OBSOLETE_LINE_FOLDING);
377 }
378 std::string headerName;
379 std::string headerValue;
380 const std::size_t delimiterPos = headerLine.find_first_of(':');
381 if (delimiterPos != std::string::npos) {
382 headerName = headerLine.substr(0, delimiterPos);
383 validateHeaderName(headerName, request);
384 webutils::lowercase(headerName);
385 headerValue = headerLine.substr(delimiterPos + 1);
386 if (headerValue[headerValue.size() - 1] == '\r')
387 headerValue.erase(headerValue.size() - 1);
388 headerValue = webutils::trimLeadingWhitespaces(headerValue);
389 webutils::trimTrailingWhiteSpaces(headerValue);
390 validateContentLength(headerName, headerValue, request);
391 validateNoMultipleHostHeaders(headerName, request);
392 request.headers[headerName] = headerValue;
393 LOG_DEBUG << "Parsed header: " << headerName << " -> " << headerValue;
394 }
395 }
396 validateHostHeader(request);
397 validateTransferEncoding(request);
398 validateMethodWithBody(request);
