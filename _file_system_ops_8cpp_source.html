50 struct stat fileStat = {};

97std::string FileSystemOps::getFileContents(const char* filename) const
98{
99 errno = 0;
100 std::ifstream fileStream(filename, std::ios::in | std::ios::binary);
101 if (!fileStream.is_open()) {
102 if (errno == ENOENT)
103 throw FileNotFoundException("std::ifstream: " + std::string(strerror(errno)));
104 if (errno == EACCES)
105 throw NoPermissionException("std::ifstream: " + std::string(strerror(errno)));

113 fileStream.read(&contents[0], static_cast<std::streamsize>(contents.size()));
