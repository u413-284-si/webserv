649bool createVirtualServer(Server& server, const std::string& host, int backlog, const std::string& port)
650{
651 struct addrinfo* list = server.resolveListeningAddresses(host, port);
652 if (list == NULL)
653 return false;
654
655 size_t successfulSock = 0;
656 size_t countTryCreateSocket = 1;
657 for (struct addrinfo* curr = list; curr != NULL; curr = curr->ai_next) {
658 LOG_DEBUG << countTryCreateSocket << ". try to create listening socket";
659
660 const int newFd = server.createListeningSocket(curr, backlog);
661 if (newFd == -1)
662 continue;
663
664 const Socket serverSock = server.retrieveSocketInfo(curr->ai_addr);
665 if (serverSock.host.empty() && serverSock.port.empty()) {
666 close(newFd);
667 continue;
668 }
669
670 if (!server.registerVirtualServer(newFd, serverSock))
671 continue;
672
673 ++successfulSock;
674 }
675 freeaddrinfo(list);
676
677 if (successfulSock == 0) {
678 LOG_ERROR << "Cannot bind to a valid socket.";
679 return false;
680 }
681
682 return true;
683}

741void handleEvent(Server& server, struct epoll_event event)
742{
743 uint32_t eventMask = event.events;
744
745 std::map<int, Socket>::const_iterator iter = server.getVirtualServers().find(event.data.fd);
746 std::map<int, Connection*>::iterator cgiIter = server.getCGIConnections().find(event.data.fd);
747 if (iter != server.getVirtualServers().end()) {
748 if ((eventMask & EPOLLERR) != 0) {
749 LOG_ERROR << "Error condition happened on the associated file descriptor of " << iter->second;
750 server.removeVirtualServer(event.data.fd);
751 return;
752 }
753 acceptConnections(server, iter->first, iter->second, eventMask);
754 } else if (cgiIter != server.getCGIConnections().end()) {
755 handleConnection(server, cgiIter->first, *cgiIter->second);
756 } else {
757 if ((eventMask & EPOLLERR) != 0) {
758 LOG_DEBUG << "epoll_wait: EPOLLERR";
759 eventMask = EPOLLIN;
760 } else if ((eventMask & EPOLLHUP) != 0) {
761 LOG_DEBUG << "epoll_wait: EPOLLHUP";
762 eventMask = EPOLLIN;
763 }
764 handleConnection(server, event.data.fd, server.getConnections().at(event.data.fd));
765 }

789void acceptConnections(Server& server, int serverFd, const Socket& serverSock, uint32_t eventMask)
790{
791 LOG_DEBUG << "Accept connections on: " << serverSock;
792
793 if ((eventMask & EPOLLIN) == 0) {
794 LOG_ERROR << "Received unknown event:" << eventMask;
795 return;
796 }
797
798 const bool isWildcardServer = (serverSock.host == "0.0.0.0");
799
800 while (true) {
801 struct sockaddr_storage clientAddr = {};
802 socklen_t clientLen = sizeof(clientAddr);
803
804 // NOLINTNEXTLINE: we need to use reinterpret_cast to convert sockaddr_storage to sockaddr
805 struct sockaddr* addrCast = reinterpret_cast<struct sockaddr*>(&clientAddr);
806
807 const int clientFd = server.acceptSingleConnection(serverFd, addrCast, &clientLen);
808 if (clientFd == -2)
809 return; // No more pending connections
810 if (clientFd == -1)
811 continue; // Error accepting connection
812
813 const Socket clientSock = server.retrieveSocketInfo(addrCast);
814 if (clientSock.host.empty() && clientSock.port.empty()) {
815 close(clientFd);
816 continue;
817 }
818
819 const Socket boundSock = isWildcardServer ? server.retrieveBoundSocketInfo(clientFd) : serverSock;
820 if (boundSock.host.empty() && boundSock.port.empty()) {
821 close(clientFd);
822 continue;
823 }
824
825 if (!server.registerConnection(boundSock, clientFd, clientSock))
826 continue;
827 }
828}

862void handleConnection(Server& server, const int activeFd, Connection& connection)
863{
864 LOG_DEBUG << "Handling connection: " << connection.m_clientSocket << " on fd: " << activeFd
865 << " for server: " << connection.m_serverSocket;
866
867 switch (connection.m_status) {
868 case (Connection::Idle):
869 connection.m_status = Connection::ReceiveHeader;
870 connectionReceiveHeader(server, activeFd, connection);
871 break;
872 case (Connection::ReceiveHeader):
873 connectionReceiveHeader(server, activeFd, connection);
874 break;
875 case (Connection::ReceiveBody):
876 connectionReceiveBody(server, activeFd, connection);
877 break;
878 case (Connection::SendToCGI):
879 connectionSendToCGI(server, connection);
880 break;
881 case (Connection::ReceiveFromCGI):
882 connectionReceiveFromCGI(server, connection);
883 break;
884 case (Connection::BuildResponse):
885 connectionBuildResponse(server, activeFd, connection);
886 break;
887 case (Connection::SendResponse):
888 connectionSendResponse(server, activeFd, connection);
889 break;
890 case (Connection::Timeout):
891 connectionHandleTimeout(server, activeFd, connection);
892 break;
893 case (Connection::Closed):
894 break;
895 }

1000void handleCompleteRequestHeader(Server& server, int clientFd, Connection& connection)
1001{
1002 LOG_DEBUG << "Received complete request header: " << '\n' << connection.m_buffer;
1003
1004 try {
1005 server.parseHeader(connection.m_buffer, connection.m_request);
1006 } catch (std::exception& e) {
1007 LOG_ERROR << e.what();
1008 connection.m_status = Connection::BuildResponse;
1009 server.modifyEvent(clientFd, EPOLLOUT);
1010 return;
1011 }
1012
1013 if (connection.m_request.contentLength > connection.location->maxBodySize) {
1014 connection.m_request.httpStatus = StatusRequestEntityTooLarge;
1015 connection.m_request.shallCloseConnection = true;
1016 connection.m_status = Connection::BuildResponse;
1017 server.modifyEvent(clientFd, EPOLLOUT);
1018 return;
1019 }
1020
1021 std::map<std::string, std::string>::iterator iter = connection.m_request.headers.find("host");
1022 if (iter != connection.m_request.headers.end()) {
1023 if (!hasValidServerConfig(connection, server.getServerConfigs(), iter->second)) {
1024 LOG_ERROR << "Failed to set active server for " << connection.m_clientSocket;
1025 server.removeConnection(clientFd);
1026 return;
1027 }
1028 }
1029 LOG_DEBUG << "Active server: " << connection.m_serverSocket;
1030
1031 server.findTargetResource(connection);
1032
1033 if (connection.m_request.hasReturn) {
1034 connection.m_status = Connection::BuildResponse;
1035 server.modifyEvent(clientFd, EPOLLOUT);
1036 return;
1037 }
1038
1039 // Allow for non-existing files when POST is used
1040 if (connection.m_request.method == MethodPost && connection.m_request.httpStatus == StatusNotFound)
1041 connection.m_request.httpStatus = StatusOK;
1042
1043 // Allow access to directories w/o autoindex when POST is used
1044 if (connection.m_request.method == MethodPost && connection.m_request.isDirectory)
1045 connection.m_request.httpStatus = StatusOK;
1046
1047 // FIXME:
1048 // Allow directories to be deleted
1049 // if (connection.m_request.method == MethodDelete && connection.m_request.isDirectory)
1050 // connection.m_request.httpStatus = StatusOK;
1051

1053 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
1054 if (!connection.location->allowMethods[connection.m_request.method])
1055 connection.m_request.httpStatus = StatusMethodNotAllowed;
1056 connection.m_request.shallCloseConnection = true;
1057
1058 if (isCGIRequested(connection)) {
1059 connection.m_request.hasCGI = true;
1060 CGIHandler cgiHandler(connection, server.getProcessOps());
1061 if (connection.m_request.httpStatus == StatusInternalServerError) {
1062 connection.m_status = Connection::BuildResponse;
1063 server.modifyEvent(clientFd, EPOLLOUT);
1064 return;
1065 }
1066 cgiHandler.execute(server.getEpollFd(), server.getConnections(), server.getCGIConnections());
1067 }
1068
1069 LOG_DEBUG << "HTTP Status: " << connection.m_request.httpStatus;
1070
1071 if (connection.m_request.httpStatus != StatusOK) {
1072 connection.m_status = Connection::BuildResponse;
1073 server.modifyEvent(clientFd, EPOLLOUT);
1074 return;
1075 }
1076
1077 if (connection.m_request.hasBody) {
1078 connection.m_status = Connection::ReceiveBody;
1079 connection.m_buffer.erase(0, connection.m_buffer.find("\r\n\r\n") + 4);
1080 if (!connection.m_buffer.empty())
1081 handleBody(server, clientFd, connection);
1082 } else if (connection.m_request.hasCGI) {
1083 connection.m_status = Connection::ReceiveFromCGI;
1084 if (!server.registerCGIFileDescriptor(connection.m_pipeFromCGIReadEnd, EPOLLIN, connection)) {
1085 connection.m_request.httpStatus = StatusInternalServerError;
1086 connection.m_status = Connection::BuildResponse;
1087 server.modifyEvent(connection.m_clientFd, EPOLLOUT);
1088 } else
1089 server.removeEvent(clientFd);
1090 } else {
1091 connection.m_status = Connection::BuildResponse;
1092 server.modifyEvent(clientFd, EPOLLOUT);
1093 }

1201void handleBody(Server& server, int activeFd, Connection& connection)
1202{
1203 if (!connection.m_request.isChunked && connection.m_request.contentLength == connection.m_buffer.size())
1204 connection.m_request.isCompleteBody = true;
1205
1206 if (connection.m_request.isChunked) {
1207 try {
1208 server.parseChunkedBody(connection.m_buffer, connection.m_request);
1209 } catch (std::exception& e) {
1210 LOG_ERROR << e.what();
1211 connection.m_status = Connection::BuildResponse;
1212 server.modifyEvent(activeFd, EPOLLOUT);
1213 return;
1214 }
1215 }
1216
1217 if (!connection.m_request.isCompleteBody) {
1218 LOG_DEBUG << "Received partial request body";
1219 // Printing body can be confusing for big files.
1220 // LOG_DEBUG << connection.m_buffer;
1221 if (connection.m_request.httpStatus != StatusOK) {
1222 connection.m_status = Connection::BuildResponse;
1223 connection.m_request.shallCloseConnection = true;
1224 server.modifyEvent(activeFd, EPOLLOUT);
1225 } else if (connection.m_buffer.size() >= connection.location->maxBodySize) {
1226 LOG_ERROR << "Maximum allowed client request body size reached from " << connection.m_clientSocket;
1227 connection.m_request.httpStatus = StatusRequestEntityTooLarge;
1228 connection.m_request.shallCloseConnection = true;
1229 connection.m_status = Connection::BuildResponse;
1230 server.modifyEvent(activeFd, EPOLLOUT);
1231 } else if (!connection.m_request.isChunked && connection.m_request.contentLength < connection.m_buffer.size()) {
1232 LOG_ERROR << ERR_CONTENT_LENGTH;
1233 LOG_ERROR << "Content-Length: " << connection.m_request.contentLength
1234 << ", Buffer size: " << connection.m_buffer.size();
1235 connection.m_request.httpStatus = StatusBadRequest;
1236 connection.m_request.shallCloseConnection = true;
1237 connection.m_status = Connection::BuildResponse;
1238 server.modifyEvent(activeFd, EPOLLOUT);
1239 }
1240 return;
1241 }
1242
1243 LOG_DEBUG << "Received complete request body";
1244 // Printing body can be confusing for big files.
1245 // LOG_DEBUG << connection.m_buffer;
1246 if (!connection.m_request.isChunked)
1247 connection.m_request.body = connection.m_buffer;
1248
1249 if (connection.m_request.hasMultipartFormdata) {
1250 try {
1251 server.decodeMultipartFormdata(connection.m_request);
1252 } catch (std::runtime_error& e) {
1253 LOG_ERROR << e.what();
1254 connection.m_request.httpStatus = StatusBadRequest;
1255 connection.m_request.shallCloseConnection = true;
1256 connection.m_status = Connection::BuildResponse;
1257 server.modifyEvent(activeFd, EPOLLOUT);
1258 return;
1259 }
1260 }
1261
1262 if (connection.m_request.hasCGI) {
1263 connection.m_status = Connection::SendToCGI;
1264 if (connection.m_request.method == MethodPost
1265 && !server.registerCGIFileDescriptor(connection.m_pipeToCGIWriteEnd, EPOLLOUT, connection)) {
1266 connection.m_request.httpStatus = StatusInternalServerError;
1267 connection.m_status = Connection::BuildResponse;
1268 server.modifyEvent(activeFd, EPOLLOUT);
1269 return;
1270 }
1271 server.removeEvent(activeFd);
1272 } else {
1273 connection.m_status = Connection::BuildResponse;
1274 server.modifyEvent(activeFd, EPOLLOUT);
1275 }

1299void connectionSendToCGI(Server& server, Connection& connection)
1300{
1301 LOG_DEBUG << "Send to CGI for: " << connection.m_clientSocket;
1302
1303 if (connection.m_request.body.empty()) {
1304 LOG_ERROR << "empty body: can't send to CGI";
1305 connection.m_request.httpStatus = StatusInternalServerError;
1306 connection.m_status = Connection::BuildResponse;
1307 server.addEvent(connection.m_clientFd, EPOLLOUT);
1308 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1309 return;
1310 }
1311
1312 const ssize_t bytesSent = server.writeProcess(
1313 connection.m_pipeToCGIWriteEnd, connection.m_request.body.c_str(), connection.m_request.body.size());
1314 LOG_DEBUG << "Bytes sent to CGI: " << bytesSent;
1315
1316 if (bytesSent == -1) {
1317 LOG_ERROR << "write(): can't send to CGI: " << std::strerror(errno);
1318 connection.m_request.httpStatus = StatusInternalServerError;
1319 connection.m_status = Connection::BuildResponse;
1320 server.addEvent(connection.m_clientFd, EPOLLOUT);
1321 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1322 return;
1323 }
1324 if (bytesSent == static_cast<ssize_t>(connection.m_request.body.size())) {
1325 LOG_DEBUG << "CGI: Full body sent";
1326 connection.m_status = Connection::ReceiveFromCGI;
1327 connection.m_request.body.clear();
1328 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1329 if (!server.registerCGIFileDescriptor(connection.m_pipeFromCGIReadEnd, EPOLLIN, connection)) {
1330 connection.m_request.httpStatus = StatusInternalServerError;
1331 connection.m_status = Connection::BuildResponse;
1332 server.addEvent(connection.m_clientFd, EPOLLOUT);
1333 }
1334 return;
1335 }
1336 connection.m_request.body.erase(0, bytesSent);
1337}

1351void connectionReceiveFromCGI(Server& server, Connection& connection)
1352{
1353 LOG_DEBUG << "Receive from CGI for: " << connection.m_clientSocket;
1354
1355 std::vector<char>& buffer = server.getBuffer();
1356 if (buffer.capacity() < Server::s_cgiBodyBufferSize)
1357 buffer.resize(Server::s_cgiBodyBufferSize);
1358
1359 const ssize_t bytesRead
1360 = server.readProcess(connection.m_pipeFromCGIReadEnd, &buffer[0], Server::s_cgiBodyBufferSize);
1361 LOG_DEBUG << "Bytes read: " << bytesRead;
1362
1363 if (bytesRead == -1) {
1364 LOG_ERROR << "read(): can't read from CGI: " << std::strerror(errno);
1365 connection.m_request.httpStatus = StatusInternalServerError;
1366 connection.m_status = Connection::BuildResponse;
1367 server.addEvent(connection.m_clientFd, EPOLLOUT);
1368 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1369 if (connection.m_pipeToCGIWriteEnd != -1)
1370 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1371 return;
1372 }
1373 if (bytesRead == 0) {
1374 LOG_DEBUG << "CGI: Full body received";
1375 connection.m_status = Connection::BuildResponse;
1376 server.addEvent(connection.m_clientFd, EPOLLOUT);
1377 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1378 if (connection.m_pipeToCGIWriteEnd != -1)
1379 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1380 int status = 0;
1381 if (server.waitForProcess(connection.m_cgiPid, &status, 0) == -1) {
1382 connection.m_request.httpStatus = StatusInternalServerError;
1383 return;
1384 }
1385 // Check if the child exited normally with exit() or returning from main()
1386 if (WIFEXITED(status)) { // NOLINT: misinterpretation by HIC++ standard
1387 int exitCode = WEXITSTATUS(status); // NOLINT: misinterpretation by HIC++ standard
1388 // Any child exit status unequal to 0 indicates unsuccessful completion of the process
1389 if (exitCode != 0) {
1390 LOG_ERROR << "child returned with: " << exitCode;
1391 connection.m_request.httpStatus = StatusInternalServerError;
1392 return;
1393 }
1394 } else if (WIFSIGNALED(status)) { // NOLINT: misinterpretation by HIC++ standard
1395 int signalNumber = WTERMSIG(status); // NOLINT: misinterpretation by HIC++ standard
1396 LOG_ERROR << "child terminated by signal: " << signalNumber << " (" << signalNumToName(signalNumber) << ")";
1397 connection.m_request.httpStatus = StatusInternalServerError;
1398 return;
1399 }
1400 }
1401 connection.m_request.body.append(buffer.begin(), buffer.begin() + bytesRead);
1402}
