659bool createVirtualServer(Server& server, const std::string& host, int backlog, const std::string& port)
660{
661 struct addrinfo* list = server.resolveListeningAddresses(host, port);
662 if (list == NULL)
663 return false;
664
665 size_t successfulSock = 0;
666 size_t countTryCreateSocket = 1;
667 for (struct addrinfo* curr = list; curr != NULL; curr = curr->ai_next) {
668 LOG_DEBUG << countTryCreateSocket << ". try to create listening socket";
669
670 const int newFd = server.createListeningSocket(curr, backlog);
671 if (newFd == -1)
672 continue;
673
674 const Socket serverSock = server.retrieveSocketInfo(curr->ai_addr);
675 if (serverSock.host.empty() && serverSock.port.empty()) {
676 close(newFd);
677 continue;
678 }
679
680 if (!server.registerVirtualServer(newFd, serverSock))
681 continue;
682
683 ++successfulSock;
684 }
685 freeaddrinfo(list);
686
687 if (successfulSock == 0) {
688 LOG_ERROR << "Cannot bind to a valid socket.";
689 return false;
690 }
691
692 return true;
693}

751void handleEvent(Server& server, struct epoll_event event)
752{
753 uint32_t eventMask = event.events;
754
755 std::map<int, Socket>::const_iterator iter = server.getVirtualServers().find(event.data.fd);
756 std::map<int, Connection*>::iterator cgiIter = server.getCGIConnections().find(event.data.fd);
757 if (iter != server.getVirtualServers().end()) {
758 if ((eventMask & EPOLLERR) != 0) {
759 LOG_ERROR << "Error condition happened on the associated file descriptor of " << iter->second;
760 server.removeVirtualServer(event.data.fd);
761 return;
762 }
763 acceptConnections(server, iter->first, iter->second, eventMask);
764 } else if (cgiIter != server.getCGIConnections().end()) {
765 handleConnection(server, cgiIter->first, *cgiIter->second);
766 } else {
767 if ((eventMask & EPOLLERR) != 0) {
768 LOG_DEBUG << "epoll_wait: EPOLLERR";
769 eventMask = EPOLLIN;
770 } else if ((eventMask & EPOLLHUP) != 0) {
771 LOG_DEBUG << "epoll_wait: EPOLLHUP";
772 eventMask = EPOLLIN;
773 }
774 handleConnection(server, event.data.fd, server.getConnections().at(event.data.fd));
775 }
776}

799void acceptConnections(Server& server, int serverFd, const Socket& serverSock, uint32_t eventMask)
800{
801 LOG_DEBUG << "Accept connections on: " << serverSock;
802
803 if ((eventMask & EPOLLIN) == 0) {
804 LOG_ERROR << "Received unknown event:" << eventMask;
805 return;
806 }
807
808 const bool isWildcardServer = (serverSock.host == "0.0.0.0");
809
810 while (true) {
811 struct sockaddr_storage clientAddr = {};
812 socklen_t clientLen = sizeof(clientAddr);
813
814 // NOLINTNEXTLINE: we need to use reinterpret_cast to convert sockaddr_storage to sockaddr
815 struct sockaddr* addrCast = reinterpret_cast<struct sockaddr*>(&clientAddr);
816
817 const int clientFd = server.acceptSingleConnection(serverFd, addrCast, &clientLen);
818 if (clientFd == -2)
819 return; // No more pending connections
820 if (clientFd == -1)
821 continue; // Error accepting connection
822
823 const Socket clientSock = server.retrieveSocketInfo(addrCast);
824 if (clientSock.host.empty() && clientSock.port.empty()) {
825 close(clientFd);
826 continue;
827 }
828
829 const Socket boundSock = isWildcardServer ? server.retrieveBoundSocketInfo(clientFd) : serverSock;
830 if (boundSock.host.empty() && boundSock.port.empty()) {
831 close(clientFd);
832 continue;
833 }
834
835 if (!server.registerConnection(boundSock, clientFd, clientSock))
836 continue;
837 }

872void handleConnection(Server& server, const int activeFd, Connection& connection)
873{
874 LOG_DEBUG << "Handling connection: " << connection.m_clientSocket << " on fd: " << activeFd
875 << " for server: " << connection.m_serverSocket;
876
877 switch (connection.m_status) {
878 case (Connection::Idle):
879 connection.m_status = Connection::ReceiveHeader;
880 connectionReceiveHeader(server, activeFd, connection);
881 break;
882 case (Connection::ReceiveHeader):
883 connectionReceiveHeader(server, activeFd, connection);
884 break;
885 case (Connection::ReceiveBody):
886 connectionReceiveBody(server, activeFd, connection);
887 break;
888 case (Connection::SendToCGI):
889 connectionSendToCGI(server, connection);
890 break;
891 case (Connection::ReceiveFromCGI):
892 connectionReceiveFromCGI(server, connection);
893 break;
894 case (Connection::BuildResponse):
895 connectionBuildResponse(server, activeFd, connection);
896 break;
897 case (Connection::SendResponse):
898 connectionSendResponse(server, activeFd, connection);
899 break;
900 case (Connection::Timeout):
901 connectionHandleTimeout(server, activeFd, connection);
902 break;
903 case (Connection::Closed):
904 break;
905 }

1009void handleCompleteRequestHeader(Server& server, int clientFd, Connection& connection)
1010{
1011 LOG_DEBUG << "Received complete request header: " << '\n' << connection.m_buffer;
1012
1013 try {
1014 server.parseHeader(connection.m_buffer, connection.m_request);
1015 } catch (const RequestParser::HTTPErrorException& e) {
1016 LOG_ERROR << e.what();
1017 connection.m_request.httpStatus = e.statusCode;
1018 connection.m_status = Connection::BuildResponse;
1019 server.modifyEvent(clientFd, EPOLLOUT);
1020 return;
1021 }
1022
1023 std::map<std::string, std::string>::iterator iter = connection.m_request.headers.find("host");
1024 if (iter != connection.m_request.headers.end()) {
1025 if (!hasValidServerConfig(connection, server.getServerConfigs(), iter->second)) {
1026 LOG_ERROR << "Failed to set active server for " << connection.m_clientSocket;
1027 server.removeConnection(clientFd);
1028 return;
1029 }
1030 }
1031 LOG_DEBUG << "Active server: " << connection.m_serverSocket;
1032
1033 server.findTargetResource(connection);
1034
1035 if (connection.m_request.contentLength > connection.location->maxBodySize) {
1036 connection.m_request.httpStatus = StatusRequestEntityTooLarge;
1037 connection.m_status = Connection::BuildResponse;
1038 server.modifyEvent(clientFd, EPOLLOUT);
1039 return;
1040 }
1041
1042 if (connection.m_request.hasReturn) {
1043 connection.m_status = Connection::BuildResponse;
1044 server.modifyEvent(clientFd, EPOLLOUT);
1045 return;
1046 }
1047
1048 // Allow for non-existing files when POST is used
1049 if (connection.m_request.method == MethodPost && connection.m_request.httpStatus == StatusNotFound)
1050 connection.m_request.httpStatus = StatusOK;
1051
1052 // Allow access to directories w/o autoindex when POST is used
1053 if (connection.m_request.method == MethodPost && connection.m_request.isDirectory)
1054 connection.m_request.httpStatus = StatusOK;
1055

1057 // Allow directories to be deleted

1063 if (!connection.location->allowMethods[connection.m_request.method]) {
1064 connection.m_request.httpStatus = StatusMethodNotAllowed;
1065 connection.m_status = Connection::BuildResponse;
1066 server.modifyEvent(clientFd, EPOLLOUT);
1067 return;
1068 }
1069
1070 if (isCGIRequested(connection)) {
1071 connection.m_request.hasCGI = true;
1072 CGIHandler cgiHandler(connection, server.getProcessOps(), server.getFileSystemOps());
1073 if (connection.m_request.httpStatus == StatusInternalServerError) {
1074 connection.m_status = Connection::BuildResponse;
1075 server.modifyEvent(clientFd, EPOLLOUT);
1076 return;
1077 }

1079 }
1080
1081 LOG_DEBUG << "HTTP Status: " << connection.m_request.httpStatus;
1082
1083 if (connection.m_request.httpStatus != StatusOK) {
1084 connection.m_status = Connection::BuildResponse;
1085 server.modifyEvent(clientFd, EPOLLOUT);
1086 return;
1087 }
1088

1090 connection.m_status = Connection::ReceiveBody;
1091 connection.m_buffer.erase(0, connection.m_buffer.find("\r\n\r\n") + 4);
1092 if (!connection.m_buffer.empty())
1093 handleBody(server, clientFd, connection);
1094 } else if (connection.m_request.hasCGI) {
1095 connection.m_status = Connection::ReceiveFromCGI;
1096 if (!server.registerCGIFileDescriptor(connection.m_pipeFromCGIReadEnd, EPOLLIN, connection)) {
1097 connection.m_request.httpStatus = StatusInternalServerError;
1098 connection.m_status = Connection::BuildResponse;
1099 server.modifyEvent(connection.m_clientFd, EPOLLOUT);
1100 } else
1101 server.removeEvent(clientFd);
1102 } else {
1103 connection.m_status = Connection::BuildResponse;
1104 server.modifyEvent(clientFd, EPOLLOUT);
1105 }

1213void handleBody(Server& server, int activeFd, Connection& connection)
1214{
1215 if (!connection.m_request.isChunked && connection.m_request.contentLength == connection.m_buffer.size())
1216 connection.m_request.isCompleteBody = true;
1217
1218 if (connection.m_request.isChunked) {
1219 try {
1220 server.parseChunkedBody(connection.m_buffer, connection.m_request);
1221 } catch (const RequestParser::HTTPErrorException& e) {
1222 LOG_ERROR << e.what();
1223 connection.m_request.httpStatus = e.statusCode;
1224 connection.m_status = Connection::BuildResponse;
1225 server.modifyEvent(activeFd, EPOLLOUT);
1226 return;
1227 }
1228 }
1229
1230 if (!connection.m_request.isCompleteBody) {
1231 LOG_DEBUG << "Received partial request body";
1232 // Printing body can be confusing for big files.
1233 // LOG_DEBUG << connection.m_buffer;
1234 if (connection.m_request.httpStatus != StatusOK) {
1235 connection.m_status = Connection::BuildResponse;
1236 server.modifyEvent(activeFd, EPOLLOUT);
1237 } else if (connection.m_buffer.size() >= connection.location->maxBodySize) {
1238 LOG_ERROR << "Maximum allowed client request body size reached from " << connection.m_clientSocket;
1239 connection.m_request.httpStatus = StatusRequestEntityTooLarge;
1240 connection.m_status = Connection::BuildResponse;
1241 server.modifyEvent(activeFd, EPOLLOUT);
1242 } else if (!connection.m_request.isChunked && connection.m_request.contentLength < connection.m_buffer.size()) {
1243 LOG_ERROR << ERR_CONTENT_LENGTH;
1244 LOG_ERROR << "Content-Length: " << connection.m_request.contentLength
1245 << ", Buffer size: " << connection.m_buffer.size();
1246 connection.m_request.httpStatus = StatusBadRequest;
1247 connection.m_status = Connection::BuildResponse;
1248 server.modifyEvent(activeFd, EPOLLOUT);
1249 }
1250 return;
1251 }
1252
1253 LOG_DEBUG << "Received complete request body";
1254 // Printing body can be confusing for big files.
1255 // LOG_DEBUG << connection.m_buffer;
1256 if (!connection.m_request.isChunked)
1257 connection.m_request.body = connection.m_buffer;
1258
1259 if (connection.m_request.hasMultipartFormdata) {
1260 try {
1261 server.decodeMultipartFormdata(connection.m_request);
1262 } catch (const RequestParser::HTTPErrorException& e) {
1263 LOG_ERROR << e.what();
1264 connection.m_request.httpStatus = e.statusCode;
1265 connection.m_status = Connection::BuildResponse;
1266 server.modifyEvent(activeFd, EPOLLOUT);
1267 return;
1268 }
1269 }
1270
1271 if (connection.m_request.hasCGI) {
1272 connection.m_status = Connection::SendToCGI;
1273 if (connection.m_request.method == MethodPost
1274 && !server.registerCGIFileDescriptor(connection.m_pipeToCGIWriteEnd, EPOLLOUT, connection)) {
1275 connection.m_request.httpStatus = StatusInternalServerError;
1276 connection.m_status = Connection::BuildResponse;
1277 server.modifyEvent(activeFd, EPOLLOUT);
1278 return;
1279 }
1280 server.removeEvent(activeFd);
1281 } else {
1282 connection.m_status = Connection::BuildResponse;
1283 server.modifyEvent(activeFd, EPOLLOUT);
1284 }

1308void connectionSendToCGI(Server& server, Connection& connection)
1309{
1310 LOG_DEBUG << "Send to CGI for: " << connection.m_clientSocket;
1311

1313 LOG_ERROR << "empty body: can't send to CGI";
1314 connection.m_request.httpStatus = StatusInternalServerError;
1315 connection.m_status = Connection::BuildResponse;
1316 server.addEvent(connection.m_clientFd, EPOLLOUT);
1317 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);

1319 }

1322 connection.m_pipeToCGIWriteEnd, connection.m_request.body.c_str(), connection.m_request.body.size());

1324
1325 if (bytesSent == -1) {
1326 LOG_ERROR << "write(): can't send to CGI: " << std::strerror(errno);
1327 connection.m_request.httpStatus = StatusInternalServerError;
1328 connection.m_status = Connection::BuildResponse;
1329 server.addEvent(connection.m_clientFd, EPOLLOUT);
1330 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1331 return;
1332 }
1333 if (bytesSent == static_cast<ssize_t>(connection.m_request.body.size())) {
1334 LOG_DEBUG << "CGI: Full body sent";
1335 connection.m_status = Connection::ReceiveFromCGI;
1336 connection.m_request.body.clear();
1337 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1338 if (!server.registerCGIFileDescriptor(connection.m_pipeFromCGIReadEnd, EPOLLIN, connection)) {
1339 connection.m_request.httpStatus = StatusInternalServerError;
1340 connection.m_status = Connection::BuildResponse;
1341 server.addEvent(connection.m_clientFd, EPOLLOUT);
1342 }
1343 return;
1344 }
1345 connection.m_request.body.erase(0, bytesSent);

1360void connectionReceiveFromCGI(Server& server, Connection& connection)
1361{
1362 LOG_DEBUG << "Receive from CGI for: " << connection.m_clientSocket;
1363
1364 std::vector<char>& buffer = server.getBuffer();
1365 if (buffer.capacity() < Server::s_cgiBodyBufferSize)
1366 buffer.resize(Server::s_cgiBodyBufferSize);
1367
1368 const ssize_t bytesRead
1369 = server.readProcess(connection.m_pipeFromCGIReadEnd, &buffer[0], Server::s_cgiBodyBufferSize);
1370 LOG_DEBUG << "Bytes read: " << bytesRead;
1371
1372 if (bytesRead == -1) {
1373 LOG_ERROR << "read(): can't read from CGI: " << std::strerror(errno);
1374 connection.m_request.httpStatus = StatusInternalServerError;
1375 connection.m_status = Connection::BuildResponse;
1376 server.addEvent(connection.m_clientFd, EPOLLOUT);
1377 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1378 if (connection.m_pipeToCGIWriteEnd != -1)
1379 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1380 return;
1381 }
1382 if (bytesRead == 0) {
1383 LOG_DEBUG << "CGI: Full body received";
1384 connection.m_status = Connection::BuildResponse;
1385 server.addEvent(connection.m_clientFd, EPOLLOUT);
1386 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1387 if (connection.m_pipeToCGIWriteEnd != -1)
1388 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1389 int status = 0;
1390 if (server.waitForProcess(connection.m_cgiPid, &status, 0) == -1) {
1391 connection.m_request.httpStatus = StatusInternalServerError;
1392 return;
1393 }
1394 connection.m_cgiPid = -1;
1395 // Check if the child exited normally with exit() or returning from main()
1396 if (WIFEXITED(status)) { // NOLINT: misinterpretation by HIC++ standard
1397 int exitCode = WEXITSTATUS(status); // NOLINT: misinterpretation by HIC++ standard
1398 // Any child exit status unequal to 0 indicates unsuccessful completion of the process
1399 if (exitCode != 0) {
1400 LOG_ERROR << "child returned with: " << exitCode;
1401 connection.m_request.httpStatus = StatusInternalServerError;
1402 return;
1403 }
1404 } else if (WIFSIGNALED(status)) { // NOLINT: misinterpretation by HIC++ standard
1405 int signalNumber = WTERMSIG(status); // NOLINT: misinterpretation by HIC++ standard
1406 LOG_ERROR << "child terminated by signal: " << signalNumber << " (" << signalNumToName(signalNumber) << ")";
1407 connection.m_request.httpStatus = StatusInternalServerError;
1408 return;
1409 }
1410 }
1411 connection.m_request.body.append(buffer.begin(), buffer.begin() + bytesRead);
1412}
