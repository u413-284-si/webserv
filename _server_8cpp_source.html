599bool createVirtualServer(Server& server, const std::string& host, int backlog, const std::string& port)
600{
601 struct addrinfo* list = server.resolveListeningAddresses(host, port);
602 if (list == NULL)
603 return false;
604
605 size_t successfulSock = 0;
606 size_t countTryCreateSocket = 1;
607 for (struct addrinfo* curr = list; curr != NULL; curr = curr->ai_next) {
608 LOG_DEBUG << countTryCreateSocket << ". try to create listening socket";
609
610 const int newFd = server.createListeningSocket(curr, backlog);
611 if (newFd == -1)
612 continue;
613
614 const Socket serverSock = server.retrieveSocketInfo(curr->ai_addr);
615 if (serverSock.host.empty() && serverSock.port.empty()) {
616 close(newFd);
617 continue;
618 }
619
620 if (!server.registerVirtualServer(newFd, serverSock))
621 continue;
622
623 ++successfulSock;
624 }
625 freeaddrinfo(list);
626
627 if (successfulSock == 0) {
628 LOG_ERROR << "Cannot bind to a valid socket.";
629 return false;
630 }
631
632 return true;
633}

692void handleEvent(Server& server, struct epoll_event event)
693{
694 uint32_t eventMask = event.events;
695
696 std::map<int, Socket>::const_iterator iter = server.getVirtualServers().find(event.data.fd);
697 std::map<int, Connection*>::iterator cgiIter = server.getCGIConnections().find(event.data.fd);
698 if (iter != server.getVirtualServers().end()) {
699 if ((eventMask & EPOLLERR) != 0) {
700 LOG_ERROR << "Error condition happened on the associated file descriptor of " << iter->second;
701 close(event.data.fd);
702 return;
703 }
704 acceptConnections(server, iter->first, iter->second, eventMask);
705 } else if (cgiIter != server.getCGIConnections().end()) {
706 handleConnection(server, cgiIter->first, *cgiIter->second);
707 } else {
708 if ((eventMask & EPOLLERR) != 0) {
709 LOG_DEBUG << "epoll_wait: EPOLLERR";
710 eventMask = EPOLLIN;
711 } else if ((eventMask & EPOLLHUP) != 0) {
712 LOG_DEBUG << "epoll_wait: EPOLLHUP";
713 eventMask = EPOLLIN;
714 }
715 handleConnection(server, event.data.fd, server.getConnections().at(event.data.fd));
716 }

740void acceptConnections(Server& server, int serverFd, const Socket& serverSock, uint32_t eventMask)
741{
742 LOG_DEBUG << "Accept connections on: " << serverSock;
743
744 if ((eventMask & EPOLLIN) == 0) {
745 LOG_ERROR << "Received unknown event:" << eventMask;
746 return;
747 }
748
749 const bool isWildcardServer = (serverSock.host == "0.0.0.0");
750
751 while (true) {
752 struct sockaddr_storage clientAddr = {};
753 socklen_t clientLen = sizeof(clientAddr);
754
755 // NOLINTNEXTLINE: we need to use reinterpret_cast to convert sockaddr_storage to sockaddr
756 struct sockaddr* addrCast = reinterpret_cast<struct sockaddr*>(&clientAddr);
757
758 const int clientFd = server.acceptSingleConnection(serverFd, addrCast, &clientLen);
759 if (clientFd == -2)
760 return; // No more pending connections
761 if (clientFd == -1)
762 continue; // Error accepting connection
763
764 const Socket clientSock = server.retrieveSocketInfo(addrCast);
765 if (clientSock.host.empty() && clientSock.port.empty()) {
766 close(clientFd);
767 continue;
768 }
769
770 const Socket boundSock = isWildcardServer ? server.retrieveBoundSocketInfo(clientFd) : serverSock;
771 if (boundSock.host.empty() && boundSock.port.empty()) {
772 close(clientFd);
773 continue;
774 }
775
776 if (!server.registerConnection(boundSock, clientFd, clientSock))
777 continue;
778 }

813void handleConnection(Server& server, const int activeFd, Connection& connection)
814{
815 LOG_DEBUG << "Handling connection: " << connection.m_clientSocket << " on fd: " << activeFd
816 << " for server: " << connection.m_serverSocket;
817
818 switch (connection.m_status) {
819 case (Connection::Idle):
820 connection.m_status = Connection::ReceiveHeader;
821 connectionReceiveHeader(server, activeFd, connection);
822 break;
823 case (Connection::ReceiveHeader):
824 connectionReceiveHeader(server, activeFd, connection);
825 break;
826 case (Connection::ReceiveBody):
827 connectionReceiveBody(server, activeFd, connection);
828 break;
829 case (Connection::SendToCGI):
830 connectionSendToCGI(server, activeFd, connection);
831 break;
832 case (Connection::ReceiveFromCGI):
833 connectionReceiveFromCGI(server, activeFd, connection);
834 break;
835 case (Connection::BuildResponse):
836 connectionBuildResponse(server, activeFd, connection);
837 break;
838 case (Connection::SendResponse):
839 connectionSendResponse(server, activeFd, connection);
840 break;
841 case (Connection::Timeout):
842 connectionHandleTimeout(server, activeFd, connection);
843 break;
844 case (Connection::Closed):
845 break;
846 }

871{

873 return;
874 LOG_DEBUG << "Receive Request Header for: " << connection.m_clientSocket;
875
876 std::vector<char>& buffer = server.getBuffer();
877 if (buffer.capacity() < Server::s_clientHeaderBufferSize)
878 buffer.resize(Server::s_clientHeaderBufferSize);
879
880 const size_t bytesToRead = Server::s_clientHeaderBufferSize - connection.m_buffer.size();
881 LOG_DEBUG << "Bytes to read: " << bytesToRead;
882
883 const ssize_t bytesRead = server.readFromSocket(activeFd, &buffer[0], bytesToRead, 0);
884 LOG_DEBUG << "Bytes read: " << bytesRead;
885

887 LOG_ERROR << "Internal server error while reading from socket: " << connection.m_clientSocket;
888 close(activeFd);

890 return;

896 return;

900 connection.m_timeSinceLastEvent = std::time(0);

902 handleCompleteRequestHeader(server, activeFd, connection);

904 LOG_DEBUG << "Received partial request header: " << '\n' << connection.m_buffer;
905 if (connection.m_buffer.size() >= Server::s_clientHeaderBufferSize) {
906 LOG_ERROR << "Buffer full, didn't receive complete request header from " << connection.m_clientSocket;

953void handleCompleteRequestHeader(Server& server, int clientFd, Connection& connection)
954{
955 LOG_DEBUG << "Received complete request header: " << '\n' << connection.m_buffer;
956
957 try {
958 server.parseHeader(connection.m_buffer, connection.m_request);
959 } catch (std::exception& e) {
960 LOG_ERROR << e.what();
961 connection.m_status = Connection::BuildResponse;
962 server.modifyEvent(clientFd, EPOLLOUT);
963 return;
964 }
965
966 std::map<std::string, std::string>::iterator iter = connection.m_request.headers.find("host");
967 if (iter != connection.m_request.headers.end()) {
968 if (!hasValidServerConfig(connection, server.getServerConfigs(), iter->second)) {
969 LOG_ERROR << "Failed to set active server for " << connection.m_clientSocket;
970 close(clientFd);
971 connection.m_status = Connection::Closed;
972 return;
973 }
974 }
975 LOG_DEBUG << "Active server: " << connection.m_serverSocket;
976
977 server.findTargetResource(connection);
978
979 if (webutils::isRedirectionStatus(connection.m_request.httpStatus))
980 connection.m_request.headers["location"] = connection.m_request.targetResource;
981
982 if (connection.m_request.hasReturn) {
983 connection.m_status = Connection::BuildResponse;
984 server.modifyEvent(clientFd, EPOLLOUT);
985 return;
986 }
987
988 // Allow for non-existing files when POST is used
989 if (connection.m_request.method == MethodPost && connection.m_request.httpStatus == StatusNotFound)
990 connection.m_request.httpStatus = StatusOK;
991
992 // Allow directories to be deleted
993 if (connection.m_request.method == MethodDelete && connection.m_request.httpStatus == StatusForbidden)
994 connection.m_request.httpStatus = StatusOK;
995
996 // bool array and method are scoped with enum Method
997 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
998 if (!connection.location->allowedMethods[connection.m_request.method]) {
999 connection.m_request.httpStatus = StatusMethodNotAllowed;
1000 connection.m_request.shallCloseConnection = true;
1001 }
1002
1003 if (isCGIRequested(connection)) {
1004 connection.m_request.hasCGI = true;
1005 ProcessOps processOps;
1006 CGIHandler cgiHandler(connection, processOps);
1007 if (connection.m_request.httpStatus == StatusInternalServerError) {
1008 connection.m_status = Connection::BuildResponse;
1009 server.modifyEvent(clientFd, EPOLLOUT);
1010 return;
1011 }
1012 cgiHandler.execute(server.getEpollFd(), server.getConnections(), server.getCGIConnections());
1013 if ((connection.m_request.method == MethodPost
1014 && !server.registerCGIFileDescriptor(connection.m_pipeToCGIWriteEnd, EPOLLOUT, connection))
1015 || !server.registerCGIFileDescriptor(connection.m_pipeFromCGIReadEnd, EPOLLIN, connection)) {
1016 connection.m_request.hasCGI = false;
1017 connection.m_request.httpStatus = StatusInternalServerError;
1018 }
1019 }
1020
1021 if (connection.m_request.httpStatus == StatusOK && connection.m_request.hasBody) {
1022 connection.m_status = Connection::ReceiveBody;
1023 connection.m_buffer.erase(0, connection.m_buffer.find("\r\n\r\n") + 4);
1024 if (!connection.m_buffer.empty())
1025 handleBody(server, clientFd, connection);
1026 } else if (connection.m_request.hasCGI)
1027 connection.m_status = Connection::ReceiveFromCGI;
1028 else {
1029 connection.m_status = Connection::BuildResponse;
1030 server.modifyEvent(clientFd, EPOLLOUT);
1031 }
1032}

1278void connectionReceiveFromCGI(Server& server, int activeFd, Connection& connection)
1279{
1280 if (activeFd == connection.m_clientFd)
1281 return;
1282
1283 LOG_DEBUG << "Receive from CGI for: " << connection.m_clientSocket;
1284
1285 std::vector<char>& buffer = server.getBuffer();
1286 if (buffer.capacity() < Server::s_cgiBodyBufferSize)
1287 buffer.resize(Server::s_cgiBodyBufferSize);
1288
1289 const ssize_t bytesRead
1290 = server.readProcess(connection.m_pipeFromCGIReadEnd, &buffer[0], Server::s_cgiBodyBufferSize);
1291 LOG_DEBUG << "Bytes read: " << bytesRead;
1292
1293 if (bytesRead == -1) {
1294 LOG_ERROR << "read(): can't read from CGI: " << std::strerror(errno);
1295 connection.m_request.httpStatus = StatusInternalServerError;
1296 connection.m_status = Connection::BuildResponse;
1297 server.modifyEvent(connection.m_clientFd, EPOLLOUT);

1299 if (connection.m_pipeToCGIWriteEnd != -1)
1300 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1301 return;
1302 }
1303 if (bytesRead == 0) {
1304 LOG_DEBUG << "CGI: Full body received";
1305 connection.m_status = Connection::BuildResponse;
1306 server.modifyEvent(connection.m_clientFd, EPOLLOUT);
1307 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1308 if (connection.m_pipeToCGIWriteEnd != -1)
1309 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1310 int status = 0;
1311 if (waitpid(connection.m_cgiPid, &status, 0) == -1) {
1312 LOG_ERROR << "waitpid(): " << std::strerror(errno);
1313 connection.m_request.httpStatus = StatusInternalServerError;
1314 return;
1315 }
1316 // Check if the child exited normally with exit() or returning from main()
1317 if (WIFEXITED(status)) { // NOLINT: misinterpretation by HIC++ standard
1318 int exitCode = WEXITSTATUS(status); // NOLINT: misinterpretation by HIC++ standard
1319 // Any child exit status unequal to 0 indicates unsuccessful completion of the process
1320 if (exitCode != 0) {
1321 LOG_ERROR << "child returned with: " << exitCode;
1322 connection.m_request.httpStatus = StatusInternalServerError;
1323 return;
1324 }
1325 } else if (WIFSIGNALED(status)) { // NOLINT: misinterpretation by HIC++ standard
1326 int signalNumber = WTERMSIG(status); // NOLINT: misinterpretation by HIC++ standard
1327 LOG_ERROR << "child terminated by signal: " << signalNumber << " (" << signalNumToName(signalNumber) << ")";
1328 connection.m_request.httpStatus = StatusInternalServerError;
1329 return;
1330 }
1331 }
1332 connection.m_request.body.append(buffer.begin(), buffer.begin() + bytesRead);
1333}
