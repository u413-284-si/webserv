613bool createVirtualServer(Server& server, const std::string& host, int backlog, const std::string& port)
614{
615 struct addrinfo* list = server.resolveListeningAddresses(host, port);
616 if (list == NULL)
617 return false;
618
619 size_t successfulSock = 0;
620 size_t countTryCreateSocket = 1;
621 for (struct addrinfo* curr = list; curr != NULL; curr = curr->ai_next) {
622 LOG_DEBUG << countTryCreateSocket << ". try to create listening socket";
623
624 const int newFd = server.createListeningSocket(curr, backlog);
625 if (newFd == -1)
626 continue;
627
628 const Socket serverSock = server.retrieveSocketInfo(curr->ai_addr);
629 if (serverSock.host.empty() && serverSock.port.empty()) {
630 close(newFd);
631 continue;
632 }
633
634 if (!server.registerVirtualServer(newFd, serverSock))
635 continue;
636
637 ++successfulSock;
638 }
639 freeaddrinfo(list);
640
641 if (successfulSock == 0) {
642 LOG_ERROR << "Cannot bind to a valid socket.";
643 return false;
644 }
645
646 return true;
647}

705void handleEvent(Server& server, struct epoll_event event)
706{
707 uint32_t eventMask = event.events;
708
709 std::map<int, Socket>::const_iterator iter = server.getVirtualServers().find(event.data.fd);
710 std::map<int, Connection*>::iterator cgiIter = server.getCGIConnections().find(event.data.fd);
711 if (iter != server.getVirtualServers().end()) {
712 if ((eventMask & EPOLLERR) != 0) {
713 LOG_ERROR << "Error condition happened on the associated file descriptor of " << iter->second;
714 close(event.data.fd);
715 return;
716 }
717 acceptConnections(server, iter->first, iter->second, eventMask);
718 } else if (cgiIter != server.getCGIConnections().end()) {
719 handleConnection(server, cgiIter->first, *cgiIter->second);
720 } else {
721 if ((eventMask & EPOLLERR) != 0) {
722 LOG_DEBUG << "epoll_wait: EPOLLERR";
723 eventMask = EPOLLIN;
724 } else if ((eventMask & EPOLLHUP) != 0) {
725 LOG_DEBUG << "epoll_wait: EPOLLHUP";
726 eventMask = EPOLLIN;
727 }
728 handleConnection(server, event.data.fd, server.getConnections().at(event.data.fd));
729 }

753void acceptConnections(Server& server, int serverFd, const Socket& serverSock, uint32_t eventMask)
754{
755 LOG_DEBUG << "Accept connections on: " << serverSock;
756
757 if ((eventMask & EPOLLIN) == 0) {
758 LOG_ERROR << "Received unknown event:" << eventMask;
759 return;
760 }
761
762 const bool isWildcardServer = (serverSock.host == "0.0.0.0");
763
764 while (true) {
765 struct sockaddr_storage clientAddr = {};
766 socklen_t clientLen = sizeof(clientAddr);
767
768 // NOLINTNEXTLINE: we need to use reinterpret_cast to convert sockaddr_storage to sockaddr
769 struct sockaddr* addrCast = reinterpret_cast<struct sockaddr*>(&clientAddr);
770
771 const int clientFd = server.acceptSingleConnection(serverFd, addrCast, &clientLen);
772 if (clientFd == -2)
773 return; // No more pending connections
774 if (clientFd == -1)
775 continue; // Error accepting connection
776
777 const Socket clientSock = server.retrieveSocketInfo(addrCast);
778 if (clientSock.host.empty() && clientSock.port.empty()) {
779 close(clientFd);
780 continue;
781 }
782
783 const Socket boundSock = isWildcardServer ? server.retrieveBoundSocketInfo(clientFd) : serverSock;
784 if (boundSock.host.empty() && boundSock.port.empty()) {
785 close(clientFd);
786 continue;
787 }
788
789 if (!server.registerConnection(boundSock, clientFd, clientSock))
790 continue;
791 }
792}

826void handleConnection(Server& server, const int activeFd, Connection& connection)
827{
828 LOG_DEBUG << "Handling connection: " << connection.m_clientSocket << " on fd: " << activeFd
829 << " for server: " << connection.m_serverSocket;
830
831 switch (connection.m_status) {
832 case (Connection::Idle):
833 connection.m_status = Connection::ReceiveHeader;
834 connectionReceiveHeader(server, activeFd, connection);
835 break;
836 case (Connection::ReceiveHeader):
837 connectionReceiveHeader(server, activeFd, connection);
838 break;
839 case (Connection::ReceiveBody):

843 connectionSendToCGI(server, activeFd, connection);
844 break;
845 case (Connection::ReceiveFromCGI):

847 break;
848 case (Connection::BuildResponse):
849 connectionBuildResponse(server, activeFd, connection);
850 break;

852 connectionSendResponse(server, activeFd, connection);
853 break;
854 case (Connection::Timeout):

964void handleCompleteRequestHeader(Server& server, int clientFd, Connection& connection)
965{
966 LOG_DEBUG << "Received complete request header: " << '\n' << connection.m_buffer;
967
968 try {
969 server.parseHeader(connection.m_buffer, connection.m_request);
970 } catch (std::exception& e) {
971 LOG_ERROR << e.what();
972 connection.m_status = Connection::BuildResponse;
973 server.modifyEvent(clientFd, EPOLLOUT);
974 return;
975 }
976
977 std::map<std::string, std::string>::iterator iter = connection.m_request.headers.find("host");
978 if (iter != connection.m_request.headers.end()) {
979 if (!hasValidServerConfig(connection, server.getServerConfigs(), iter->second)) {
980 LOG_ERROR << "Failed to set active server for " << connection.m_clientSocket;
981 server.removeConnection(clientFd);
982 return;
983 }
984 }
985 LOG_DEBUG << "Active server: " << connection.m_serverSocket;
986
987 server.findTargetResource(connection);
988
989 if (webutils::isRedirectionStatus(connection.m_request.httpStatus))
990 connection.m_request.headers["location"] = connection.m_request.targetResource;
991
992 if (connection.m_request.hasReturn) {
993 connection.m_status = Connection::BuildResponse;
994 server.modifyEvent(clientFd, EPOLLOUT);
995 return;
996 }
997
998 // Allow for non-existing files when POST is used
999 if (connection.m_request.method == MethodPost && connection.m_request.httpStatus == StatusNotFound)
1000 connection.m_request.httpStatus = StatusOK;
1001
1002 // Allow directories to be deleted
1003 if (connection.m_request.method == MethodDelete && connection.m_request.httpStatus == StatusForbidden)
1004 connection.m_request.httpStatus = StatusOK;
1005
1006 // bool array and method are scoped with enum Method
1007 // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-constant-array-index)
1008 if (!connection.location->allowedMethods[connection.m_request.method]) {
1009 connection.m_request.httpStatus = StatusMethodNotAllowed;
1010 connection.m_request.shallCloseConnection = true;
1011 }
1012
1013 if (isCGIRequested(connection)) {
1014 connection.m_request.hasCGI = true;
1015 ProcessOps processOps;
1016 CGIHandler cgiHandler(connection, processOps);
1017 if (connection.m_request.httpStatus == StatusInternalServerError) {
1018 connection.m_status = Connection::BuildResponse;
1019 server.modifyEvent(clientFd, EPOLLOUT);
1020 return;
1021 }
1022 cgiHandler.execute(server.getEpollFd(), server.getConnections(), server.getCGIConnections());
1023 if ((connection.m_request.method == MethodPost
1024 && !server.registerCGIFileDescriptor(connection.m_pipeToCGIWriteEnd, EPOLLOUT, connection))
1025 || !server.registerCGIFileDescriptor(connection.m_pipeFromCGIReadEnd, EPOLLIN, connection)) {
1026 connection.m_request.hasCGI = false;
1027 connection.m_request.httpStatus = StatusInternalServerError;
1028 }
1029 }
1030
1031 if (connection.m_request.httpStatus == StatusOK && connection.m_request.hasBody) {
1032 connection.m_status = Connection::ReceiveBody;
1033 connection.m_buffer.erase(0, connection.m_buffer.find("\r\n\r\n") + 4);
1034 if (!connection.m_buffer.empty())
1035 handleBody(server, clientFd, connection);
1036 } else if (connection.m_request.hasCGI)
1037 connection.m_status = Connection::ReceiveFromCGI;
1038 else {
1039 connection.m_status = Connection::BuildResponse;
1040 server.modifyEvent(clientFd, EPOLLOUT);
1041 }
1042}

1286void connectionReceiveFromCGI(Server& server, int activeFd, Connection& connection)
1287{
1288 if (activeFd == connection.m_clientFd)
1289 return;
1290
1291 LOG_DEBUG << "Receive from CGI for: " << connection.m_clientSocket;
1292
1293 std::vector<char>& buffer = server.getBuffer();
1294 if (buffer.capacity() < Server::s_cgiBodyBufferSize)
1295 buffer.resize(Server::s_cgiBodyBufferSize);
1296
1297 const ssize_t bytesRead
1298 = server.readProcess(connection.m_pipeFromCGIReadEnd, &buffer[0], Server::s_cgiBodyBufferSize);
1299 LOG_DEBUG << "Bytes read: " << bytesRead;
1300
1301 if (bytesRead == -1) {
1302 LOG_ERROR << "read(): can't read from CGI: " << std::strerror(errno);
1303 connection.m_request.httpStatus = StatusInternalServerError;
1304 connection.m_status = Connection::BuildResponse;
1305 server.modifyEvent(connection.m_clientFd, EPOLLOUT);
1306 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1307 if (connection.m_pipeToCGIWriteEnd != -1)
1308 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1309 return;
1310 }
1311 if (bytesRead == 0) {
1312 LOG_DEBUG << "CGI: Full body received";
1313 connection.m_status = Connection::BuildResponse;
1314 server.modifyEvent(connection.m_clientFd, EPOLLOUT);
1315 server.removeCGIFileDescriptor(connection.m_pipeFromCGIReadEnd);
1316 if (connection.m_pipeToCGIWriteEnd != -1)
1317 server.removeCGIFileDescriptor(connection.m_pipeToCGIWriteEnd);
1318 int status = 0;
1319 if (waitpid(connection.m_cgiPid, &status, 0) == -1) {
1320 LOG_ERROR << "waitpid(): " << std::strerror(errno);
1321 connection.m_request.httpStatus = StatusInternalServerError;
1322 return;
1323 }
1324 // Check if the child exited normally with exit() or returning from main()
1325 if (WIFEXITED(status)) { // NOLINT: misinterpretation by HIC++ standard
1326 int exitCode = WEXITSTATUS(status); // NOLINT: misinterpretation by HIC++ standard
1327 // Any child exit status unequal to 0 indicates unsuccessful completion of the process
1328 if (exitCode != 0) {
1329 LOG_ERROR << "child returned with: " << exitCode;
1330 connection.m_request.httpStatus = StatusInternalServerError;
1331 return;
1332 }
1333 } else if (WIFSIGNALED(status)) { // NOLINT: misinterpretation by HIC++ standard
1334 int signalNumber = WTERMSIG(status); // NOLINT: misinterpretation by HIC++ standard
1335 LOG_ERROR << "child terminated by signal: " << signalNumber << " (" << signalNumToName(signalNumber) << ")";
1336 connection.m_request.httpStatus = StatusInternalServerError;
1337 return;
1338 }
1339 }
1340 connection.m_request.body.append(buffer.begin(), buffer.begin() + bytesRead);
1341}
